scripted_gui = {
    building_construction_speeds_gui = {
        context_type = player_context
        window_name = "MD_building_construction_speeds"
        parent_window_token = construction_tab

        visible = {
            check_variable = { MD_constructionview_tab_selected = 0 }
        }
    }
    countryinvestmentsview_tabs = {
		window_name = "MD_countryinvestmentsview_tabs"
		context_type = player_context
		parent_window_token = construction_tab

		triggers = {
			construction_tab_click_enabled = {
				if = {
					limit = { check_variable = { MD_constructionview_tab_selected = 0 } }
					set_temp_variable = { construction_tab_frame = 1 }
				}
				else = {
					set_temp_variable = { construction_tab_frame = 2 }
				}

				always = yes
			}

			investments_tab_click_enabled = {
				if = {
					limit = { check_variable = { MD_constructionview_tab_selected = 1 } }
					set_temp_variable = { investments_tab_frame = 1 }
				}
				else = {
					set_temp_variable = { investments_tab_frame = 2 }
				}

				always = yes
			}
		}

		effects = {
			construction_tab_click = {
				set_variable = { MD_constructionview_tab_selected = 0 }
			}

			investments_tab_click = {
				set_variable = { MD_constructionview_tab_selected = 1 }
                set_variable = { MD_investments_selected_building = -1 }
                update_dirty_investments = yes
			}
		}

		properties = {
			construction_tab = {
				frame = construction_tab_frame
			}

			investments_tab = {
				frame = investments_tab_frame
			}
		}
	}
    countryinvestmentsview = {
		window_name = "MD_countryinvestmentsview"
		context_type = player_context
		parent_window_token = construction_tab

        dirty = investments_dirty

        visible = {
			is_ai = no
			check_variable = { MD_constructionview_tab_selected = 1 }
		}

        triggers = {
            select_civ_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 0 }
			}

			select_mil_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 1 }
			}

			select_dockyard_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 2 }
			}

			select_infrastructure_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 3 }
			}

			select_office_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 4 }
			}

			select_antyair_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 5 }
			}

			select_radar_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 6 }
			}

			select_airbase_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 7 }
			}

			select_strategic_resource_reserve_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 8 }
			}

			select_internet_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 9 }
			}

			select_renewable_power_plant_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 10 }
			}
			
			select_fossil_fuel_power_plant_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 11 }
			}

			select_nuclear_power_plant_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 12 }
			}

			select_agriculture_overlay_glow_visible = {
				check_variable = { MD_investments_selected_building = 13 }
			}

            # investment grid
            increase_priority_click_enabled = {
                check_variable = { i > 0 }
            }

            ongoing_contruction_anim_visible = {
                check_variable = { i < MD_investments_max_projects_base }
            }

            decrease_priority_click_enabled = {
                set_temp_variable = { max_index = investments_projects^num }
                subtract_from_temp_variable = { max_index = 1 }
                check_variable = { i < max_index }
            }
            
            increase_amount_click_enabled = {
                set_temp_variable = { proggres_frame = investments_projects_building_time_max^v }
                subtract_from_temp_variable = { proggres_frame = investments_projects_building_time^v }
                divide_temp_variable = { proggres_frame = investments_projects_building_time_max^v }
                multiply_temp_variable = { proggres_frame = 100 }
                round_variable = proggres_frame

                always = yes
            }
        }

        effects = {
            select_civ_button_click = {
                set_variable = { MD_investments_selected_building = 0 }
	            update_dirty_investments = yes
            }
            select_mil_button_click = {
                set_variable = { MD_investments_selected_building = 1 }
	            update_dirty_investments = yes
            }
            select_dockyard_button_click = {
                set_variable = { MD_investments_selected_building = 2 }
	            update_dirty_investments = yes
            }
            select_infrastructure_button_click = {
                set_variable = { MD_investments_selected_building = 3 }
	            update_dirty_investments = yes
            }
            select_office_button_click = {
                set_variable = { MD_investments_selected_building = 4 }
	            update_dirty_investments = yes
            }
            select_antyair_button_click = {
                set_variable = { MD_investments_selected_building = 5 }
	            update_dirty_investments = yes
            }
            select_radar_button_click = {
                set_variable = { MD_investments_selected_building = 6 }
	            update_dirty_investments = yes
            }
            select_airbase_button_click = {
                set_variable = { MD_investments_selected_building = 7 }
	            update_dirty_investments = yes
            }
            select_strategic_resource_reserve_button_click = {
                set_variable = { MD_investments_selected_building = 8 }
	            update_dirty_investments = yes
            }
            select_internet_button_click = {
                set_variable = { MD_investments_selected_building = 9 }
	            update_dirty_investments = yes
            }
            select_renewable_power_plant_button_click = {
                set_variable = { MD_investments_selected_building = 10 }
	            update_dirty_investments = yes
            }
            select_fossil_fuel_power_plant_button_click = {
                set_variable = { MD_investments_selected_building = 11 }
	            update_dirty_investments = yes
            }
            select_nuclear_power_plant_button_click = {
                set_variable = { MD_investments_selected_building = 12 }
	            update_dirty_investments = yes
            }
            select_agriculture_button_click = {
                set_variable = { MD_investments_selected_building = 13 }
	            update_dirty_investments = yes
            }

            # investment grid
            remove_investment_click = {
                set_temp_variable = { building_index = investments_projects_building^v }
                set_temp_variable = { amount = investments_projects_building_amount^v }
                set_temp_variable = { state = investments_projects_state^v }
                remove_investment_project = yes
            }

            increase_priority_click = {
                set_temp_variable = { index = i }
                add_to_temp_variable = { index = -1 }
                set_temp_variable = { index_to_switch = investments_projects^index }

                set_variable = { investments_projects^index = v }
                set_variable = { investments_projects^i = index_to_switch }
            }

            decrease_priority_click = {
                set_temp_variable = { index = i }
                add_to_temp_variable = { index = 1 }
                set_temp_variable = { index_to_switch = investments_projects^index }

                set_variable = { investments_projects^index = v }
                set_variable = { investments_projects^i = index_to_switch }
            }

            increase_amount_click = {
                set_temp_variable = { state = investments_projects_state^v }
                    set_temp_variable = { building_index = investments_projects_building^v }
                    set_temp_variable = { amount = 1 }
                    create_investment_project = yes
            }

            decrease_amount_click = {
                set_temp_variable = { state = investments_projects_state^v }
                set_temp_variable = { building_index = investments_projects_building^v }
                set_temp_variable = { amount = 1 }
                remove_investment_project = yes
            }
        }

        dynamic_lists = {
            investments_list_grid = {
                array = investments_projects
                entry_container = "investments_list_item"
            }
            
            foreign_investments_list_grid = {
                array = foreign_investments_countries
                entry_container = "foreign_investments_list_item"
            }
        }

        properties = {
			investment_building_icon = {
				frame = global.build_const_icon_array^investments_projects_building^v
			}

            investment_progress {
                frame = proggres_frame
            }
        }
    }
    investmentsmapicons = {
		context_type = state_mapicon
		window_name = "MD_investments_mapicons"

		visible = { 
            ROOT = {
                is_ai = no
                check_variable = { MD_investments_selected_building > -1 }
                check_variable = { MD_constructionview_tab_selected = 1 }
            }
        }

        triggers = {
            build_button_visible = {
                AND = {
                    NOT = { is_owned_by = ROOT }
                    NOT = { is_controlled_by = ROOT }
                }
            }

            current_projects_visible = {
                AND = {
                    NOT = { is_owned_by = ROOT }
                    NOT = { is_controlled_by = ROOT }
                }
            }

            build_button_click_enabled = {
                free_building_slots = {
                    building = arms_factory
                    size > 0
                }
            }
        }

        effects = {
            build_button_click = {
                set_temp_variable = { state = THIS }
                ROOT = {
                    set_temp_variable = { building_index = MD_investments_selected_building }
                    set_temp_variable = { amount = 1 }
                    create_investment_project = yes
                }
            }
        }
    }
}
