# Greek Creditor System Triggers

GRE_is_eligible_creditor = {
	custom_trigger_tooltip = {
		tooltip = GRE_eligible_creditor_tt
		NOT = { has_war_with = GRE }
		is_subject = no
		has_opinion = { target = GRE value > -50 }
		NOT = { is_in_array = { GRE.GRE_creditors = THIS.id } }
	}
}

GRE_has_creditors = {
	check_variable = { GRE_creditors^num > 0 }
}

GRE_is_creditor_of_greece = {
	is_in_array = { GRE.GRE_creditors = THIS.id }
}

# Unified debt crisis threshold: 150% debt-to-GDP
# Factors generic debt (full) + creditor debt (half weight)
GRE_debt_crisis_threshold = {
	custom_trigger_tooltip = {
		tooltip = GRE_debt_crisis_threshold_tt
		check_variable = { GRE_unified_debt_ratio > 1.5 }
	}
}

# Debt is manageable (below 100% unified ratio)
GRE_debt_manageable = {
	custom_trigger_tooltip = {
		tooltip = GRE_debt_manageable_tt
		check_variable = { GRE_unified_debt_ratio < 1 }
	}
}
