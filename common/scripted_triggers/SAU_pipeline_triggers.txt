# Pipeline route gates for the Saudi pipeline mini-game
SAU_saddam_in_power = {
	country_exists = IRQ
	IRQ = {
		has_government = nationalist
		NOT = { has_country_flag = IRQ_saddam_caught }
	}
}

SAU_saddam_fell = {
	NOT = { SAU_saddam_in_power = yes }
}

SAU_yemen_route_secure = {
	YEM = { controls_state = 194}
	NOT = { 194 = { CONTROLLER = { original_tag = HOU } } }
}

SAU_houthis_won_south = {
	OR = {
		194 = { CONTROLLER = { original_tag = HOU } }
		199 = { CONTROLLER = { original_tag = HOU } }
		198 = { CONTROLLER = { original_tag = HOU } }
	}
}

SAU_fsa_controls_syria = {
	country_exists = FSA
	FSA = { controls_state = 186 }
}

SAU_assad_controls_syria = {
	country_exists = SYR
	SYR = { controls_state = 186 }
}

SAU_yemen_at_war = {
	OR = {
		YEM = { has_war = yes }
		HOU = { has_war = yes }
	}
}

SAU_is_pipeline_saboteur = {
	OR = {
		original_tag = PER
		original_tag = HEZ
		original_tag = HOU
		original_tag = ISR
	}
}
