# Who is allowed to actually become a NATO member. Europe, North America and the
# Caucasus, plus the handful of scripted exceptions (Central Asian pivots, Algeria,
# and a fully reformed Russia). Anyone outside this set is redirected to MNNA status
# by NATO_join / NATO_join_via_event instead of joining.
can_join_NATO = {
	OR = {
		is_european_nation = yes
		is_North_America = yes
		is_Caucasus = yes
		original_tag = KAZ
		original_tag = UZB
		original_tag = KYR
		original_tag = ALG
		original_tag = SOV
		original_tag = TUR
		original_tag = ISR
	}
}
