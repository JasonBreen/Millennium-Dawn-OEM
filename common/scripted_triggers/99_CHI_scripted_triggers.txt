### Chinese Trigger Conditions

is_sco = {
	OR = {
		has_idea = CHI_sco_member
		has_idea = CHI_sco_member_econ
		has_idea = CHI_sco_member_pol
		has_idea = CHI_sco_member_mil
	}
}

CHI_sco_is_dialogue_partner = {
	has_idea = CHI_sco_dialogue_partner
}

CHI_sco_is_observer = {
	has_idea = CHI_sco_observer
}

CHI_sco_is_any = {
	OR = {
		has_idea = CHI_sco_dialogue_partner
		has_idea = CHI_sco_observer
		has_idea = CHI_sco_member
		has_idea = CHI_sco_member_econ
		has_idea = CHI_sco_member_pol
		has_idea = CHI_sco_member_mil
	}
}

CHI_sco_cohesion_is_high = {
	CHI = { check_variable = { CHI_sco_cohesion > 6 } }
}

CHI_sco_cohesion_is_low = {
	CHI = { check_variable = { CHI_sco_cohesion < 3 } }
}

CHI_sco_china_dominant = {
	CHI = { check_variable = { CHI_sco_balance > 3 } }
}

CHI_sco_russia_dominant = {
	CHI = { check_variable = { CHI_sco_balance < -3 } }
}

CHI_sco_can_apply_to_join = {
	has_global_flag = CHI_sco_formed
	NOT = { CHI_sco_is_any = yes }
	NOT = { is_in_faction_with = USA }
	NOT = { is_in_faction_with = ENG }
	CHI = { has_war = no }
	NOT = { has_war_with = CHI }
}

all_sco_members_are_china_ruling_party = {
	if = {
		limit = { has_government = communism }
		custom_trigger_tooltip = {
			tooltip = SCO_all_emerging_tt
			NOT = {
				any_of_scopes = {
					array = global.sco_members
					is_sco = yes
					NOT = { has_government = communism }
				}
			}
		}
	}
	if = {
		limit = { has_government = democratic }
		custom_trigger_tooltip = {
			tooltip = SCO_all_democratic_tt
			NOT = {
				any_of_scopes = {
					array = global.sco_members
					is_sco = yes
					NOT = { has_government = democratic }
				}
			}
		}
	}
}
### CHI-NKO ###

CHI_NKO_joint_focus_available = {
	CHI = {
		exists = yes
		NOT  = { has_war_with = NKO }
		has_opinion = {
			target = NKO
			value > 24
		}
	}
	NKO = {
		exists = yes
		has_opinion = {
			target = CHI
			value > 24
		}
	}
}

CHI_NKO_idea_cancel = {
	OR = {
		NOT = { country_exists = CHI }
		NOT = { country_exists = NKO }
		AND = { original_tag = CHI  has_war_with = NKO }
		AND = { original_tag = NKO  has_war_with = CHI }
	}
}
