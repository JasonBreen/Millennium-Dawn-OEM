# Joint Focus Availability Triggers — China bilateral relationships
# Used in available = { } and joint_trigger = { } blocks of CHI-NKO, CHI-SOV, and CHI-SAR shared focus trees
# Cancel triggers for ideas are also defined here

### CHI-NKO ###

CHI_NKO_joint_focus_available = {
	country_exists = CHI
	country_exists = NKO
	NOT = { has_war_with = NKO }
	NOT = { has_war_with = CHI }
	if = {
		limit = { original_tag = CHI }
		has_opinion = { target = NKO value > -1 }
	}
	if = {
		limit = { original_tag = NKO }
		has_opinion = { target = CHI value > -1 }
	}
}

CHI_NKO_idea_cancel = {
	OR = {
		NOT = { country_exists = CHI }
		NOT = { country_exists = NKO }
		AND = { original_tag = CHI  has_war_with = NKO }
		AND = { original_tag = NKO  has_war_with = CHI }
	}
}

### CHI-SAR (general — any active SAR of CHI) ###

CHI_any_SAR_joint_focus_available = {
	country_exists = CHI
	NOT = { has_war_with = CHI }
	OR = {
		original_tag = CHI
		AND = {
			OR = {
				original_tag = HKG
				original_tag = MAC
				original_tag = TIB
				original_tag = ETK
				original_tag = TAI
				original_tag = MON
			}
			is_subject_of = CHI
			if = {
				limit = { NOT = { original_tag = CHI } }
				has_opinion = { target = CHI value > -1 }
			}
		}
	}
}

### CHI-HKG ###

}
