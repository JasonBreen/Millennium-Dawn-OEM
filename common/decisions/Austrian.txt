AUS_toll_system_category = {
	AUS_toll_system = {
		fire_only_once = no

		custom_cost_trigger = {
			check_variable = { toll_cost < 10 }
		}

		complete_effect = {
			hidden_effect = {
				set_variable = { attractiveness = 5}
                set_variable = { toll_cost = 5 }
                set_variable = { toll_factors = 1 }
				calc_toll_cost = yes
			}
		}
	}

	AUS_test_increase_infra = {
		fire_only_once = no

		custom_cost_trigger = {
			check_variable = { toll_cost < 10 }
		}

		complete_effect = {
				add_to_variable = { attractiveness = 1}
				calc_toll_cost = yes
		}	
	}

	AUS_test_increase_toll_cost = {
		fire_only_once = no

		custom_cost_trigger = {
			check_variable = { toll_cost < 10 }
		}

		complete_effect = {
				add_to_variable = { toll_cost = 1}
				calc_toll_cost = yes
		}	
	}

	AUS_test_increase_toll_factors = {
		fire_only_once = no

		custom_cost_trigger = {
			check_variable = { toll_cost < 10 }
		}

		complete_effect = {
				set_variable = { toll_factors = 1.5}
				calc_toll_cost = yes
		}	
	}
}
