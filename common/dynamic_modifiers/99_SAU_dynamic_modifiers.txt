# One consolidated "national spirit" for the Salman + MBS political tree.
# Each focus that used to grant its own spirit now adds to these variables; the
# modifier shows as a single entry whose values grow as the reign is built.
SAU_royal_reign = {
	enable = { always = yes }
	icon = GFX_idea_saudi_royal_family
	custom_modifier_tooltip = SAU_royal_reign_tt
	remove_trigger = { NOT = { has_country_flag = SAU_royal_reign_active } }

	political_power_factor = SAU_reign_pp
	stability_factor = SAU_reign_stab
	war_support_factor = SAU_reign_ws
	drift_defence_factor = SAU_reign_drift
	production_speed_buildings_factor = SAU_reign_build
	consumer_goods_factor = SAU_reign_cg
}
