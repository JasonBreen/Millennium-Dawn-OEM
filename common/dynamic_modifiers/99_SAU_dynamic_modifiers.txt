# One consolidated "national spirit" for the Salman + MBS political tree.
# Each focus that used to grant its own spirit now adds to these variables; the
# modifier shows as a single entry whose values grow as the reign is built.

	enable = { always = yes }
	icon = GFX_idea_saudi_royal_family
	custom_modifier_tooltip = SAU_royal_reign_tt
	remove_trigger = { NOT = { has_country_flag = SAU_royal_reign_active } }

	political_power_factor = SAU_reign_pp
	stability_factor = SAU_reign_stab
	war_support_factor = SAU_reign_ws

	production_speed_buildings_factor = SAU_reign_build

# The Communist sub-tree's shared buffs. enable ties them to EITHER communist party in
# power: the Saudi Communist Party (ruling_party = 4) or the National Liberation Front
# (ruling_party = 19). The moment neither holds power, every buff switches off.
SAU_communist_state = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			OR = {
				emerging_communist_state_are_in_power = yes
				neutrality_neutral_communism_are_in_power = yes
			}
		}
	}
	icon = GFX_idea_CZE_communist_policy
	custom_modifier_tooltip = SAU_communist_state_tt

	political_power_factor = SAU_com_pp
	stability_factor = SAU_com_stab
	war_support_factor = SAU_com_ws
	production_speed_buildings_factor = SAU_com_build
	production_factory_max_efficiency_factor = SAU_com_factory
	consumer_goods_factor = SAU_com_cg
}
