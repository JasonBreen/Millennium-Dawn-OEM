# Called in the spawning of a new nation in-game
# Purely used to update the variables of a nation if you are trying to play as a releasable
ingame_init_investment_system = {
	set_variable = { MD_investments_max_projects = 15 }
}

create_investment_project = {
    set_temp_variable = { added = 0 }

    for_each_loop = {
        array = investments_projects_state
        index = i
        value = v

        if = {
            limit = {
                check_variable = { v = state }
                check_variable = { investments_projects_building^i = building_index }
            }
            
            add_to_variable = { investments_projects_building_amount^i = amount }

            set_temp_variable = { added = 1 }

            calculate_investment_project_cost = yes
            set_temp_variable = { treasury_change = investment_project_cost }
            multiply_temp_variable = { treasury_change = amount }
            multiply_temp_variable = { treasury_change = -1 }
            modify_treasury_effect = yes
        }
    }

    if = {
        limit = {
            check_variable = { added = 0 }
        }

        add_to_array = { investments_projects = investments_projects^num }
        add_to_array = { investments_projects_state = state }

        add_to_array = { investments_projects_building = building_index }
        add_to_array = { investments_projects_building_amount = amount }

        calculate_investment_project_time = yes
        add_to_array = { investments_projects_building_time = project_time }
        add_to_array = { investments_projects_building_time_max = project_time }

        calculate_investment_project_cost = yes
        add_to_array = { investments_projects_building_cost = investment_project_cost }

        set_temp_variable = { treasury_change = investment_project_cost }
        multiply_temp_variable = { treasury_change = amount }
        multiply_temp_variable = { treasury_change = -1 }
        modify_treasury_effect = yes
    }
}

remove_investment_project = {
    for_each_loop = {
        array = investments_projects_state
        index = i
        value = v
        break = break

        if = {
            limit = {
                check_variable = { v = state }
                check_variable = { investments_projects_building^i = building_index }
            }
            
            subtract_from_variable = { investments_projects_building_amount^i = amount }

            if = {
                limit = { check_variable = { investments_projects_building_time^i > 0 } }

                # TODO: something broken is here?
                set_temp_variable = { remove_return_procentage = investments_projects_building_time^i }
                divide_temp_variable = { remove_return_procentage = investments_projects_building_time_max^i }
            
                set_temp_variable = { treasury_change = investments_projects_building_cost^i }
                multiply_temp_variable = { treasury_change = remove_return_procentage }
                multiply_temp_variable = { treasury_change = amount }
                modify_treasury_effect = yes
            }

            calculate_investment_project_cost = yes
            set_variable = { investments_projects_building_cost^i = investment_project_cost }
            
            calculate_investment_project_time = yes
            set_variable = { investments_projects_building_time^i = project_time }
            set_variable = { investments_projects_building_time_max^i = project_time }
            
            if = {
                limit = {
                    check_variable = { investments_projects_building_amount^i < 1 }
                }

                remove_from_array = { 
                    array = investments_projects
                    value = i
                }

                remove_from_array = { 
                    array = investments_projects_building
                    index = i
                }
                remove_from_array = { 
                    array = investments_projects_building_amount
                    index = i
                }
                remove_from_array = { 
                    array = investments_projects_building_cost
                    index = i
                }

                remove_from_array = { 
                    array = investments_projects_building_time
                    index = i

    }
}

add_investment = {
    set_temp_variable = { added = 0 }

    for_each_loop = {
        array = foreign_investments_countries
        index = i
        value = v

        if = {
            limit = {
                check_variable = { v = country }
            }
            
            add_to_variable = { foreign_investments^i = amount }

            set_temp_variable = { added = 1 }
        }
    }

    if = {
        limit = {
            check_variable = { added = 0 }
        }

        add_to_array = { foreign_investments_countries = country }
        add_to_array = { foreign_investments = amount }
    }

    var:country_from = {
        add_to_variable = { int_investments = amount }
    }

    add_to_variable = { int_foreign_investments = amount }
}

update_dirty_investments = {
    if = { limit = { check_variable = { THIS.investments_dirty < 100000 } }
        add_to_variable = { THIS.investments_dirty = 1 }
    }
    else = {
        set_variable = { THIS.investments_dirty = 0 }
    }
}

# Calculate duration for a single investment project entry (new array-based system).
# Expects local vars: state, building_index, amount (set by create_investment_project).
# Reads: local_state_modifier, receiving_investment_duration_temp (precompute_state_construction_data).
# Writes: temp_variable project_time (consumed by create_investment_project).
calculate_investment_project_time = {
	set_temp_variable = { project_type = building_index }
	get_building_construction_values = yes

	# local_state_modifier precomputed by precompute_state_construction_data (per state)
	add_to_temp_variable = { construction_speed = local_state_modifier }

	# receiving_investment_duration_temp precomputed by precompute_state_construction_data
	add_to_temp_variable = { receiving_investment_duration_temp = modifier@investment_duration_modifier }
	clamp_temp_variable = { var = receiving_investment_duration_temp min = -1 }
	if = { limit = { check_variable = { receiving_investment_duration_temp > 0 } }
		add_to_temp_variable = { construction_speed = receiving_investment_duration_temp }
	}
	else = {
		subtract_from_temp_variable = { construction_speed = receiving_investment_duration_temp }
	}

	set_variable = { displayable_construction_modifier = construction_speed }

	# Assume 5 × 15 Civilian Industry devoted to a single construction line
	multiply_temp_variable = { construction_speed = 75 }
	if = { limit = { check_variable = { construction_speed > 0 } }
		divide_temp_variable = { building_cost = construction_speed }
	}
	else = {
		# construction_speed <= 0 -> clamp to 1 day to avoid instant completion
		set_temp_variable = { building_cost = 1 }
	}

	set_temp_variable = { project_time = building_cost }
	round_temp_variable = project_time
}
# Calculate treasury cost for a single investment project entry (new array-based system).
# Expects local vars: state, building_index, amount (set by create_investment_project).
# Scope: PREV = state, ROOT = investing country (unchanged from old system).
# Writes/returns: temp_variable investment_project_cost (the monetary cost).
calculate_investment_project_cost = {
	set_temp_variable = { _cost_idx = building_index }
	set_temp_variable = { investment_project_cost = global.build_cost_array^_cost_idx }

	# AC Build Cost Modifiers
	set_temp_variable = { investments_cost_modifier = 1 }
	PREV = {
		CONTROLLER = {
			add_to_temp_variable = { investments_cost_modifier = THIS.modifier@receiving_investment_cost_modifier }
			if = {
				limit = {
					ROOT = {
						OR = {
							has_idea = full_isr_boycotte
							has_idea = econ_isr_boycotte
						}
					}
				}
				add_to_temp_variable = { investments_cost_modifier = 0.5 }
			}
			if = {
				limit = {
					is_in_array = { global.EU_member = THIS.id }
					PREV.PREV = { has_idea = ECB_president }
				}
				add_to_temp_variable = { investments_cost_modifier = -0.15 }
			}
			if = {
				limit = {
					is_in_array = { global.EU_member = THIS.id }
					PREV.PREV = { has_idea = EU_finance_minister }
				}
				add_to_temp_variable = { investments_cost_modifier = -0.05 }
			}
			if = {
				limit = {
					has_idea = FRA_idea_development_investment
					PREV.PREV = { original_tag = FRA }
				}
				add_to_temp_variable = { investments_cost_modifier = -0.05 }
			}
			if = {
				limit = {
					PREV.PREV = {
						has_country_flag = mutual_investment_treaty_@ROOT
					}
				}
				add_to_temp_variable = { investments_cost_modifier = -0.05 }
			}
		}
	}
	add_to_temp_variable = { investments_cost_modifier = ROOT.modifier@investment_cost_modifier }

	# Calculates modified Build Cost
	set_variable = { displayable_monetary_cost_modifier = investments_cost_modifier }
	multiply_temp_variable = { var = investment_project_cost value = investments_cost_modifier }
	multiply_temp_variable = { var = investment_project_cost value = amount }
}
