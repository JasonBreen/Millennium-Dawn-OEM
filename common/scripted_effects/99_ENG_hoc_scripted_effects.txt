# ============================================================
# SAVE AND ZERO BLOCKED PARTY POPULARITY
# Only Military (22) and Monarchist (23) are blocked from HoC seats
# All other parties can earn seats if they have popularity
# ============================================================
ENG_hoc_save_pops = {
	# Save and zero only parties that can never hold HoC seats
	set_variable = { ENG_hoc_saved_pop^22 = party_pop_array^22 }
	set_variable = { ENG_hoc_saved_pop^23 = party_pop_array^23 }
	set_variable = { party_pop_array^22 = 0 }  # Military Junta - never electable
	set_variable = { party_pop_array^23 = 0 }  # Monarchists - never electable
}

# ============================================================
# RESTORE PARTY POPULARITY AFTER CALCULATION
# ============================================================
ENG_hoc_restore_pops = {
	set_variable = { party_pop_array^22 = ENG_hoc_saved_pop^22 }
	set_variable = { party_pop_array^23 = ENG_hoc_saved_pop^23 }
}

# ============================================================
# FIND TOP 3 PARTIES (FPTP tripartite system for England)
# Extends PER_majlis_duopoly to 3 parties
# ============================================================
ENG_hoc_tripartite = {
	# Find 1st most popular
	find_highest_in_array = {
		array = party_pop_array
		value = eng_pop_1_temp
		index = eng_idx_1_temp
	}
	set_variable = { eng_top_1 = eng_idx_1_temp }
	set_variable = { eng_save_pop_1 = party_pop_array^eng_top_1 }
	set_variable = { party_pop_array^eng_top_1 = -2 }

	# Find 2nd most popular
	find_highest_in_array = {
		array = party_pop_array
		value = eng_pop_2_temp
		index = eng_idx_2_temp
	}
	set_variable = { eng_top_2 = eng_idx_2_temp }
	set_variable = { eng_save_pop_2 = party_pop_array^eng_top_2 }
	set_variable = { party_pop_array^eng_top_2 = -2 }

	# Find 3rd most popular
	find_highest_in_array = {
		array = party_pop_array
		value = eng_pop_3_temp
		index = eng_idx_3_temp
	}
	set_variable = { eng_top_3 = eng_idx_3_temp }

	# Restore
	set_variable = { party_pop_array^eng_top_1 = eng_save_pop_1 }
	set_variable = { party_pop_array^eng_top_2 = eng_save_pop_2 }
}

# ============================================================
# ENGLAND SEATS (533)
# Without voter reform: FPTP bonuses for top 3, penalties for rest
# With voter reform: proportional
# ============================================================
ENG_hoc_england_seats = {
	if = {
		limit = { NOT = { has_country_flag = ENG_voter_reform } }
		# FPTP mode - find top 3 parties
		ENG_hoc_tripartite = yes

		for_each_loop = {
			array = party_pop_array
			value = v
			index = i

			set_variable = { ENG_hoc_seats^i = 0 }

			# Top party gets FPTP winner bonus
			if = {
				limit = { check_variable = { i = eng_top_1 } }
				set_variable = { ENG_hoc_seats^i = party_pop_array^i }
				multiply_variable = { ENG_hoc_seats^i = 1.4 }
				multiply_variable = { ENG_hoc_seats^i = 533 }
				round_variable = ENG_hoc_seats^i
			}
			# 2nd party gets mild bonus
			else_if = {
				limit = { check_variable = { i = eng_top_2 } }
				set_variable = { ENG_hoc_seats^i = party_pop_array^i }
				multiply_variable = { ENG_hoc_seats^i = 1.1 }
				multiply_variable = { ENG_hoc_seats^i = 533 }
				round_variable = ENG_hoc_seats^i
			}
			# 3rd party gets squeezed (Lib Dem effect)
			else_if = {
				limit = { check_variable = { i = eng_top_3 } }
				set_variable = { ENG_hoc_seats^i = party_pop_array^i }
				multiply_variable = { ENG_hoc_seats^i = 0.7 }
				multiply_variable = { ENG_hoc_seats^i = 533 }
				round_variable = ENG_hoc_seats^i
			}
			# Other parties with >3% popularity get heavily penalized
			else_if = {
				limit = { check_variable = { party_pop_array^i > 0.03 } }
				set_variable = { ENG_hoc_seats^i = party_pop_array^i }
				multiply_variable = { ENG_hoc_seats^i = 0.15 }
				multiply_variable = { ENG_hoc_seats^i = 533 }
				round_variable = ENG_hoc_seats^i
			}
			# Parties below 3% get nothing in FPTP

			add_to_variable = { ENG_hoc_england_total = ENG_hoc_seats^i }
		}
	}
	else = {
		# Proportional mode (voter reform passed)
		for_each_loop = {
			array = party_pop_array
			value = v
			index = i

			set_variable = { ENG_hoc_seats^i = 0 }

			if = {
				limit = { check_variable = { party_pop_array^i > 0.005 } }
				set_variable = { ENG_hoc_seats^i = party_pop_array^i }
				multiply_variable = { ENG_hoc_seats^i = 533 }
				round_variable = ENG_hoc_seats^i
			}

			add_to_variable = { ENG_hoc_england_total = ENG_hoc_seats^i }
		}
	}
	ENG_hoc_england_correct = yes
}

# ============================================================
# ENGLAND SEAT CORRECTION - ensures exactly 533
# Mirrors PER_majlis_correct_seats
# ============================================================
ENG_hoc_england_correct = {
	for_each_loop = {
		array = party_pop_array
		value = v
		index = i

		# Under 533 - add seats to parties that have at least 1
		if = {
			limit = {
				check_variable = { ENG_hoc_england_total < 534 }
				check_variable = { ENG_hoc_seats^i > 1 }
			}
			set_temp_variable = { eng_correction_needed = ENG_hoc_england_total }
			subtract_from_temp_variable = { eng_correction_needed = 534 }
			multiply_temp_variable = { eng_correction_needed = -1 }
			set_temp_variable_to_random = {
				var = eng_correction_amount
				max = eng_correction_needed
				integer = yes
			}
			add_to_variable = { ENG_hoc_seats^i = eng_correction_amount }
			add_to_variable = { ENG_hoc_england_total = eng_correction_amount }
		}
		# Over 533 - remove seats from parties that have more than 1
		else_if = {
			limit = {
				check_variable = { ENG_hoc_england_total > 532 }
				check_variable = { ENG_hoc_seats^i > 1 }
			}
			set_temp_variable = { eng_correction_needed = ENG_hoc_england_total }
			subtract_from_temp_variable = { eng_correction_needed = 532 }
			set_temp_variable_to_random = {
				var = eng_correction_amount
				max = eng_correction_needed
				integer = yes
			}
			subtract_from_variable = { ENG_hoc_seats^i = eng_correction_amount }
			subtract_from_variable = { ENG_hoc_england_total = eng_correction_amount }
		}
	}
	# Final clamp - assign remainder to independents if needed
	set_variable = { ENG_hoc_eng_remainder = 533 }
	subtract_from_variable = { ENG_hoc_eng_remainder = ENG_hoc_england_total }
	if = {
		limit = { check_variable = { ENG_hoc_eng_remainder > 0 } }
		add_to_variable = { ENG_hoc_independent_seats = ENG_hoc_eng_remainder }
		add_to_variable = { ENG_hoc_england_total = ENG_hoc_eng_remainder }
	}
}

# ============================================================
# SCOTLAND SEATS (59)
# SNP strength driven by ENG_scottish_nationalists variable
# Right-wing parties heavily penalized
# ============================================================
ENG_hoc_scotland_seats = {
	# Calculate SNP share
	set_temp_variable = { snp_share = ENG_scottish_nationalists }
	multiply_temp_variable = { snp_share = 0.01 }

	# Euroscepticism bonus: leaving EU boosts SNP
	if = {
		limit = { has_country_flag = ENG_left_eu }
		add_to_temp_variable = { snp_share = 0.05 }
	}

	# High agitation bonus
	if = {
		limit = { check_variable = { ENG_scottish_agitation > 50 } }
		set_temp_variable = { scot_agitation_bonus = ENG_scottish_agitation }
		multiply_temp_variable = { scot_agitation_bonus = 0.002 }
		add_to_temp_variable = { snp_share = scot_agitation_bonus }
	}

	# Clamp SNP share
	clamp_temp_variable = { var = snp_share min = 0.05 max = 0.85 }

	# Calculate SNP seats
	set_variable = { ENG_hoc_snp_seats = snp_share }
	multiply_variable = { ENG_hoc_snp_seats = 59 }
	round_variable = ENG_hoc_snp_seats

	# Remaining seats for UK-wide parties
	set_temp_variable = { scot_remaining = 59 }
	subtract_from_temp_variable = { scot_remaining = ENG_hoc_snp_seats }

	# Calculate weighted shares for remaining parties
	set_temp_variable = { scot_weight_total = 0 }

	for_each_loop = {
		array = party_pop_array
		value = v
		index = i

		set_temp_variable = { scot_party_weight^i = 0 }

		if = {
			limit = { check_variable = { party_pop_array^i > 0.01 } }
			set_temp_variable = { scot_party_weight^i = party_pop_array^i }

			# Right-wing penalty (Con, Reactionaries, RWP, Fascists)
			if = {
				limit = {
					OR = {
						check_variable = { i = 1 }
						check_variable = { i = 6 }
						check_variable = { i = 20 }
						check_variable = { i = 21 }
					}
				}
				multiply_temp_variable = { scot_party_weight^i = 0.3 }
			}
			# Left-wing bonus (Labour, Lib Dem, Greens)
			else_if = {
				limit = {
					OR = {
						check_variable = { i = 3 }
						check_variable = { i = 2 }
						check_variable = { i = 17 }
					}
				}
				multiply_temp_variable = { scot_party_weight^i = 1.2 }
			}

			add_to_temp_variable = { scot_weight_total = scot_party_weight^i }
		}
	}

	# Distribute remaining seats proportionally by weight
	if = {
		limit = { check_variable = { scot_weight_total > 0 } }
		for_each_loop = {
			array = party_pop_array
			value = v
			index = i

			if = {
				limit = { check_variable = { scot_party_weight^i > 0 } }
				set_temp_variable = { scot_party_share = scot_party_weight^i }
				divide_temp_variable = { scot_party_share = scot_weight_total }
				multiply_temp_variable = { scot_party_share = scot_remaining }
				round_temp_variable = scot_party_share
				add_to_variable = { ENG_hoc_seats^i = scot_party_share }
				add_to_variable = { ENG_hoc_scotland_total = scot_party_share }
			}
		}
	}
	add_to_variable = { ENG_hoc_scotland_total = ENG_hoc_snp_seats }
}

# NORTHERN IRELAND SEATS (18)
# STV (default) or FPTP (old system)
# 4 parties: DUP, UUP (unionist), Sinn Fein, SDLP (nationalist)
# ============================================================
ENG_hoc_ni_seats = {
	if = {
		limit = { NOT = { has_country_flag = ENG_ni_fptp } }
		# STV mode (post-Good Friday Agreement)
		set_temp_variable = { ni_nationalist_share = ENG_north_irish_nationalists }
		multiply_temp_variable = { ni_nationalist_share = 0.01 }

		# Ireland doing well boosts nationalists
		if = {
			limit = {
				country_exists = IRE
				IRE = { check_variable = { gdp_per_capita > 50 } }
			}
			add_to_temp_variable = { ni_nationalist_share = 0.05 }
		}

		# Clamp
		clamp_temp_variable = { var = ni_nationalist_share min = 0.25 max = 0.65 }

		# Calculate nationalist seats
		set_temp_variable = { ni_nationalist_seats = ni_nationalist_share }
		multiply_temp_variable = { ni_nationalist_seats = 18 }
		round_temp_variable = ni_nationalist_seats

		# Sinn Fein gets 55%, SDLP gets 45%
		set_variable = { ENG_hoc_sf_seats = ni_nationalist_seats }
		multiply_variable = { ENG_hoc_sf_seats = 0.55 }
		round_variable = ENG_hoc_sf_seats

		set_variable = { ENG_hoc_sdlp_seats = ni_nationalist_seats }
		subtract_from_variable = { ENG_hoc_sdlp_seats = ENG_hoc_sf_seats }

		# Unionist seats
		set_temp_variable = { ni_unionist_seats = 18 }
		subtract_from_temp_variable = { ni_unionist_seats = ni_nationalist_seats }

		# DUP gets 60%, UUP gets 40%
		set_variable = { ENG_hoc_dup_seats = ni_unionist_seats }
		multiply_variable = { ENG_hoc_dup_seats = 0.60 }
		round_variable = ENG_hoc_dup_seats

		set_variable = { ENG_hoc_uup_seats = ni_unionist_seats }
		subtract_from_variable = { ENG_hoc_uup_seats = ENG_hoc_dup_seats }

		# Random variance (+/- 1 seat between parties in each bloc)
		set_temp_variable_to_random = {
			var = ni_variance
			min = -1
			max = 2
			integer = yes
		}
		if = {
			limit = {
				check_variable = { ni_variance > 0 }
				check_variable = { ENG_hoc_sdlp_seats > 1 }
			}
			add_to_variable = { ENG_hoc_sf_seats = 1 }
			subtract_from_variable = { ENG_hoc_sdlp_seats = 1 }
		}
		else_if = {
			limit = {
				check_variable = { ni_variance < 0 }
				check_variable = { ENG_hoc_sf_seats > 1 }
			}
			subtract_from_variable = { ENG_hoc_sf_seats = 1 }
			add_to_variable = { ENG_hoc_sdlp_seats = 1 }
		}
	}
	else = {
		# FPTP mode (old system - unionist dominance)
		set_variable = { ENG_hoc_dup_seats = 10 }
		set_variable = { ENG_hoc_uup_seats = 6 }
		set_variable = { ENG_hoc_sf_seats = 1 }
		set_variable = { ENG_hoc_sdlp_seats = 1 }
		# Reignite the troubles
		if = {
			limit = { NOT = { has_country_flag = ENG_ni_troubles_reignited } }
			set_country_flag = ENG_ni_troubles_reignited
		}
	}

	# Sum NI total
	set_variable = { ENG_hoc_ni_total = ENG_hoc_dup_seats }
	add_to_variable = { ENG_hoc_ni_total = ENG_hoc_uup_seats }
	add_to_variable = { ENG_hoc_ni_total = ENG_hoc_sf_seats }
	add_to_variable = { ENG_hoc_ni_total = ENG_hoc_sdlp_seats }
}

# ============================================================
# DETERMINE GOVERNMENT
# Find largest party, check majority (326+), check extremist
# ============================================================
ENG_hoc_determine_government = {
	# Find the largest party overall (election winner)
	set_variable = { ENG_hoc_largest_party = 0 }
	set_variable = { ENG_hoc_largest_party_seats = 0 }
	for_each_loop = {
		array = party_pop_array
		value = v
		index = i
		if = {
			limit = { check_variable = { ENG_hoc_seats^i > ENG_hoc_largest_party_seats } }
			set_variable = { ENG_hoc_largest_party_seats = ENG_hoc_seats^i }
			set_variable = { ENG_hoc_largest_party = i }
		}
	}

	# Check if SNP has more seats than any main party
	if = {
		limit = { check_variable = { ENG_hoc_snp_seats > ENG_hoc_largest_party_seats } }
		set_variable = { ENG_hoc_largest_party_seats = ENG_hoc_snp_seats }
		set_variable = { ENG_hoc_largest_party = 100 } # 100 = SNP special code
	}

	# Resolve current ruling party index from the ruling_party array
	# ruling_party is an array, so we must use is_in_array checks
	set_variable = { ENG_hoc_ruling_party_index = -1 }
	if = { limit = { is_in_array = { ruling_party = 0 } } set_variable = { ENG_hoc_ruling_party_index = 0 } }
	if = { limit = { is_in_array = { ruling_party = 1 } } set_variable = { ENG_hoc_ruling_party_index = 1 } }
	if = { limit = { is_in_array = { ruling_party = 2 } } set_variable = { ENG_hoc_ruling_party_index = 2 } }
	if = { limit = { is_in_array = { ruling_party = 3 } } set_variable = { ENG_hoc_ruling_party_index = 3 } }
	if = { limit = { is_in_array = { ruling_party = 4 } } set_variable = { ENG_hoc_ruling_party_index = 4 } }
	if = { limit = { is_in_array = { ruling_party = 5 } } set_variable = { ENG_hoc_ruling_party_index = 5 } }
	if = { limit = { is_in_array = { ruling_party = 6 } } set_variable = { ENG_hoc_ruling_party_index = 6 } }
	if = { limit = { is_in_array = { ruling_party = 13 } } set_variable = { ENG_hoc_ruling_party_index = 13 } }
	if = { limit = { is_in_array = { ruling_party = 14 } } set_variable = { ENG_hoc_ruling_party_index = 14 } }
	if = { limit = { is_in_array = { ruling_party = 16 } } set_variable = { ENG_hoc_ruling_party_index = 16 } }
	if = { limit = { is_in_array = { ruling_party = 17 } } set_variable = { ENG_hoc_ruling_party_index = 17 } }
	if = { limit = { is_in_array = { ruling_party = 18 } } set_variable = { ENG_hoc_ruling_party_index = 18 } }
	if = { limit = { is_in_array = { ruling_party = 19 } } set_variable = { ENG_hoc_ruling_party_index = 19 } }
	if = { limit = { is_in_array = { ruling_party = 20 } } set_variable = { ENG_hoc_ruling_party_index = 20 } }
	if = { limit = { is_in_array = { ruling_party = 21 } } set_variable = { ENG_hoc_ruling_party_index = 21 } }

	# Get the current RULING party's seats using the resolved index
	set_variable = { ENG_hoc_ruling_seats = ENG_hoc_seats^ENG_hoc_ruling_party_index }

	# Check if ruling party won the election (is the largest party)
	set_variable = { ENG_hoc_ruling_party_won = 0 }
	if = {
		limit = { check_variable = { ENG_hoc_largest_party = ENG_hoc_ruling_party_index } }
		set_variable = { ENG_hoc_ruling_party_won = 1 }
	}

	# Majority is based on RULING party's seats, not largest party overall
	if = {
		limit = { check_variable = { ENG_hoc_ruling_seats > 325 } }
		set_variable = { ENG_hoc_majority = 1 }
	}
	else = {
		set_variable = { ENG_hoc_majority = 0 }
	}

	# Check if largest party is extremist (indices 4, 5, 19, 20, 21)
	# Only matters if monarchy still exists
	if = {
		limit = {
			NOT = { has_country_flag = ENG_monarchy_abolished }
			OR = {
				check_variable = { ENG_hoc_largest_party = 4 }
				check_variable = { ENG_hoc_largest_party = 5 }
				check_variable = { ENG_hoc_largest_party = 19 }
				check_variable = { ENG_hoc_largest_party = 20 }
				check_variable = { ENG_hoc_largest_party = 21 }
			}
		}
		set_variable = { ENG_hoc_extremist_winner = 1 }
		set_variable = { ENG_hoc_extremist_index = ENG_hoc_largest_party }
	}
}

# ============================================================
# BY-ELECTION CALCULATION
# Transfers a single seat from over-represented to under-represented party
# ============================================================
ENG_hoc_by_election_calc = {
	# Find most over-represented party (seat share > pop share)
	set_variable = { ENG_hoc_by_election_loser = -1 }
	set_variable = { ENG_hoc_by_election_winner = -1 }
	set_temp_variable = { by_election_max_over = 0 }
	set_temp_variable = { by_election_max_under = 0 }

	for_each_loop = {
		array = party_pop_array
		value = v
		index = i

		if = {
			limit = { check_variable = { ENG_hoc_seats^i > 0 } }
			# Calculate seat share
			set_temp_variable = { by_seat_share = ENG_hoc_seats^i }
			divide_temp_variable = { by_seat_share = ENG_hoc_total_seats }
			# Calculate over-representation
			set_temp_variable = { by_over_rep = by_seat_share }
			subtract_from_temp_variable = { by_over_rep = party_pop_array^i }

			if = {
				limit = { check_variable = { by_over_rep > by_election_max_over } }
				set_temp_variable = { by_election_max_over = by_over_rep }
				set_variable = { ENG_hoc_by_election_loser = i }
			}
		}

		# Under-represented (pop share > seat share)
		if = {
			limit = { check_variable = { party_pop_array^i > 0.02 } }
			set_temp_variable = { by_seat_share_2 = ENG_hoc_seats^i }
			divide_temp_variable = { by_seat_share_2 = ENG_hoc_total_seats }
			set_temp_variable = { by_under_rep = party_pop_array^i }
			subtract_from_temp_variable = { by_under_rep = by_seat_share_2 }

			if = {
				limit = { check_variable = { by_under_rep > by_election_max_under } }
				set_temp_variable = { by_election_max_under = by_under_rep }
				set_variable = { ENG_hoc_by_election_winner = i }
			}
		}
	}

	# Transfer one seat
	if = {
		limit = {
			check_variable = { ENG_hoc_by_election_loser > -1 }
			check_variable = { ENG_hoc_by_election_winner > -1 }
			NOT = { check_variable = { ENG_hoc_by_election_loser = ENG_hoc_by_election_winner } }
		}
		subtract_from_variable = { ENG_hoc_seats^ENG_hoc_by_election_loser = 1 }
		add_to_variable = { ENG_hoc_seats^ENG_hoc_by_election_winner = 1 }
	}
}
