# Pipeline tier-management effects (SAU pipeline feature)

SAU_yemen_set_tier = {
	if = { limit = { has_idea = SAU_yemen_pipeline_t1 } remove_ideas = SAU_yemen_pipeline_t1 }
	if = { limit = { has_idea = SAU_yemen_pipeline_t2 } remove_ideas = SAU_yemen_pipeline_t2 }
	if = { limit = { has_idea = SAU_yemen_pipeline_t3 } remove_ideas = SAU_yemen_pipeline_t3 }
	if = { limit = { country_exists = YEM } YEM = {
		if = { limit = { has_idea = SAU_pipeline_transit_YEM_t1 } remove_ideas = SAU_pipeline_transit_YEM_t1 }
		if = { limit = { has_idea = SAU_pipeline_transit_YEM_t2 } remove_ideas = SAU_pipeline_transit_YEM_t2 }
		if = { limit = { has_idea = SAU_pipeline_transit_YEM_t3 } remove_ideas = SAU_pipeline_transit_YEM_t3 }
	} }
	if = { limit = { check_variable = { SAU_yemen_tier = 1 } }
		add_ideas = SAU_yemen_pipeline_t1
		if = { limit = { country_exists = YEM } YEM = { add_ideas = SAU_pipeline_transit_YEM_t1 } }
	}
	if = { limit = { check_variable = { SAU_yemen_tier = 2 } }
		add_ideas = SAU_yemen_pipeline_t2
		if = { limit = { country_exists = YEM } YEM = { add_ideas = SAU_pipeline_transit_YEM_t2 } }
	}
	if = { limit = { check_variable = { SAU_yemen_tier = 3 } }
		add_ideas = SAU_yemen_pipeline_t3
		if = { limit = { country_exists = YEM } YEM = { add_ideas = SAU_pipeline_transit_YEM_t3 } }
	}
}

SAU_yemen_tier_down = {
	if = { limit = { has_idea = SAU_yemen_pipeline_t3 } set_temp_variable = { SAU_yemen_tier = 2 } SAU_yemen_set_tier = yes }
	else_if = { limit = { has_idea = SAU_yemen_pipeline_t2 } set_temp_variable = { SAU_yemen_tier = 1 } SAU_yemen_set_tier = yes }
	else_if = { limit = { has_idea = SAU_yemen_pipeline_t1 }
		set_temp_variable = { SAU_yemen_tier = 0 } SAU_yemen_set_tier = yes
		clr_country_flag = SAU_yemen_pipeline_open
		if = { limit = { has_idea = SAU_pipeline_empire } remove_ideas = SAU_pipeline_empire }
	}
}

SAU_yemen_tier_up = {
	if = { limit = { has_idea = SAU_yemen_pipeline_t1 } set_temp_variable = { SAU_yemen_tier = 2 } SAU_yemen_set_tier = yes }
	else_if = { limit = { has_idea = SAU_yemen_pipeline_t2 } set_temp_variable = { SAU_yemen_tier = 3 } SAU_yemen_set_tier = yes }
}
