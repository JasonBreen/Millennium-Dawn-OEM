# Effect: update_focus_tree_obsolete_branches
# Purpose: Updates the focus tree to hide branches that are no longer available via the game for optimizing the Focus Tree windows.
update_focus_tree_obsolete_branches = {
	hidden_effect = {
		if = {
			limit = {
				has_game_rule = {
					rule = obsolete_focus_branches_visibility
					option = HIDE
				}
			}
			mark_focus_tree_layout_dirty = yes
		}
	}
}
