# CAC_add_company
# Required Input: cac_selected_company (int)
# Required Input: cac_company_type (int)

# Optional Inputs:
# cac_company_size (int)
# cac_company_value (float)
# cac_company_money (float)
# cac_company_factories (int)

# Purpose: Add the cac_selected_company to companys list
CAC_add_company = {
    if = {
        limit = {
            check_variable = { CAC_companies^num < 11  }
        }

        custom_effect_tooltip = CAC_add_company_tt

        set_temp_variable = { company_slot_temp = CAC_companies^num }

        add_to_array = {
		    array = CAC_companies
			value = cac_selected_company
			index = company_slot_temp
		}

        add_to_array = {
		    array = CAC_companies_type
			value = cac_company_type
			index = company_slot_temp
		}

        add_to_array = {
            array = CAC_companies_size
            value = cac_company_size
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_value
            value = cac_company_value
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_money
            value = 0
            index = company_slot_temp
        }

        set_temp_variable = { cac_skip_update = 1 }

        set_temp_variable = { cac_money_change = 5 }
        set_temp_variable = { cac_slot = company_slot_temp }
        CAC_modify_money_effect = yes

        add_to_array = {
            array = CAC_companies_revenue
            value = 0
            index = company_slot_temp
        }

        if = {
            limit = {
                NOT = { check_variable = { cac_company_additional_income = 0 } }
            }

            add_to_array = {
                array = CAC_companies_additional_income
                value = cac_company_additional_income
                index = company_slot_temp
            }
        }
        else = {
            add_to_array = {
                array = CAC_companies_additional_income
                value = 0
                index = company_slot_temp
            }
        }

        if = {
            limit = {
                NOT = { check_variable = { cac_company_additional_expenses = 0 } }
            }

            add_to_array = {
                array = CAC_companies_additional_expenses
                value = cac_company_additional_expenses
                index = company_slot_temp
            }
        }
        else = {
            add_to_array = {
                array = CAC_companies_additional_expenses
                value = 0
                index = company_slot_temp
            }
        }

        add_to_array = {
			array = CAC_companies_income
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
			array = CAC_companies_expenses
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
			array = CAC_companies_taxes
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
            array = CAC_companies_display_expenses
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_maintenance
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_workforce
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_sector_buildings
            value = 0
            index = company_slot_temp
        }

        # company modifiers
        add_to_array = {
            array = CAC_companies_income_modifier
            value = 0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_expenses_modifier
            value = 0
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_modifier_from_size
            value = 0
            index = company_slot_temp
        }

        set_temp_variable = { cac_skip_update = 0 }
        CAC_update_variables = yes

        # upgrades
        if = {
            limit = { check_variable = { CAC_companies_upgrades_2d^num = 0 } }

            for_loop_effect = {
                start = 0
                end = 11
                value = company_index

                for_loop_effect = {
                    start = 0
                    end = 30
                    value = upgrade_id

                    set_temp_variable = { index = company_index }
                    multiply_temp_variable = { index = 30 }
                    add_to_temp_variable = { index = upgrade_id }

                    add_to_array = {
                        array = CAC_companies_upgrades_2d
                        value = 3
                        index = index
                    }
                }
            }
        }

        if = {
            limit = { check_variable = { cac_company_type = 0 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 1 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 2 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 3 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 4 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 5 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 6 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }

    }

# CAC_modify_money_effect
# Required Input: cac_money_change (float)
# Required Input: cac_slot (int)
# Purpose: Change company money on slot cac_slot by cac_money_change
CAC_modify_money_effect = {
    if = {
        limit = {
            AND = {
                check_variable = { cac_slot > -1 }
                check_variable = { cac_slot < CAC_companies^num }
            }
        }

        custom_effect_tooltip = CAC_modify_money_tt

        add_to_variable = { CAC_companies_money^cac_slot = cac_money_change }
        add_to_variable = { CAC_companies_value^cac_slot = cac_money_change }

        CAC_update_variables = yes
    }
}

    if = {
        limit = {
            AND = {
                check_variable = { cac_slot > -1 }
                check_variable = { cac_slot < CAC_companies^num }
            }
        }

        set_temp_variable = { index = cac_slot }
		multiply_temp_variable = { index = 30 }

		set_variable = { CAC_companies_upgrades_2d^index = cac_new_status }

        if = {
            limit = { check_variable = { cac_new_status = 1 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_1_tt
        }
        else_if = {
            limit = { check_variable = { cac_new_status = 2 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_2_tt
        }
        else_if = {
            limit = { check_variable = { cac_new_status = 3 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_3_tt
        }
    }
}

# CAC_complete_upgrade_effect
# Required Input: cac_base_money_change (int)
# Required Input: cac_slot (int)
# Purpose: Remove money from company on upgrade
CAC_complete_upgrade_effect = {
    set_temp_variable = { cac_money_change_modifier = 1 }
    add_to_temp_variable = { cac_money_change_modifier = CAC_companies_modifier_from_size^cac_slot }

    multiply_temp_variable = { cac_base_money_change = cac_money_change_modifier }

    add_to_variable = { CAC_companies_money^cac_slot = cac_base_money_change }
    add_to_variable = { CAC_companies_value^cac_slot = cac_base_money_change }

    CAC_update_variables = yes
}

# CAC_update_building_stats
# Purpose: Computes building-count-derived values for all 7 company slots:
# maintenance, workforce (stored as negative), and sector_buildings.
# Called monthly and at startup (before CAC_update_variables).
# These values only change when construction completes, so monthly cadence is sufficient.
CAC_update_building_stats = {
	# Pre-compute shared workforce scaling factor (gdp_per_capita * 0.007) used by all 7 slots
	set_temp_variable = { cac_gdpc_scaled = gdp_per_capita }
	multiply_temp_variable = { cac_gdpc_scaled = 0.007 }

	# Slot 0 ‚Äî Banking & Finance (offices)
	set_variable = { CAC_companies_maintenance^0 = office_park_total }
	multiply_variable = { CAC_companies_maintenance^0 = 0.01260 }         # 12.00 * 0.15 * 0.007
	multiply_variable = { CAC_companies_maintenance^0 = -1 }
	set_temp_variable = { cac_workforce_cost = office_park_total }
	multiply_temp_variable = { cac_workforce_cost = 0.630 }               # workers_per_office
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	set_variable = { CAC_companies_workforce^0 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^0 = -1 }
	set_variable = { CAC_companies_sector_buildings^0 = office_park_total }

	# Slot 1 ‚Äî Mining & Resources (synthetic refineries + agriculture + resource extraction)
	set_variable = { CAC_companies_maintenance^1 = synthetic_refinery_total }
	multiply_variable = { CAC_companies_maintenance^1 = 0.00630 }         # 6.00 * 0.15 * 0.007
	set_temp_variable = { cac_agri_maintenance = agriculture_district_total }
	multiply_temp_variable = { cac_agri_maintenance = 0.00525 }           # 5.00 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^1 = cac_agri_maintenance }
	multiply_variable = { CAC_companies_maintenance^1 = -1 }
	set_temp_variable = { cac_workforce_cost = synthetic_refinery_total }
	multiply_temp_variable = { cac_workforce_cost = 0.245 }               # workers_per_synthetic_refinery
	multiply_temp_variable = { cac_workforce_cost = synthetic_refinery_manpower_fulfillment }
	set_temp_variable = { cac_agri_workforce = agriculture_district_total }
	multiply_temp_variable = { cac_agri_workforce = 0.250 }               # workers_per_agriculture_district
	multiply_temp_variable = { cac_agri_workforce = agriculture_district_manpower_fulfillment }
	add_to_temp_variable = { cac_workforce_cost = cac_agri_workforce }
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	# Add resource extraction workers (already in millions, GDP-adjusted; scale by wage factor)
	set_temp_variable = { cac_resource_workforce_cost = resource_sector_workers }
	multiply_temp_variable = { cac_resource_workforce_cost = resource_sector_worker_fulfillment }
	multiply_temp_variable = { cac_resource_workforce_cost = cac_gdpc_scaled }
	add_to_temp_variable = { cac_workforce_cost = cac_resource_workforce_cost }
	set_variable = { CAC_companies_workforce^1 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^1 = -1 }
	set_variable = { CAC_companies_sector_buildings^1 = synthetic_refinery_total }
	add_to_variable = { CAC_companies_sector_buildings^1 = agriculture_district_total }

	# Slot 2 ‚Äî Energy & Utilities (fuel silos + fossil powerplants + nuclear reactors + renewable energy)
	set_variable = { CAC_companies_maintenance^2 = fuel_silos }
	multiply_variable = { CAC_companies_maintenance^2 = 0.00315 }         # 3.00 * 0.15 * 0.007
	set_temp_variable = { cac_energy_maint = number_of_fossil_pps }
	multiply_temp_variable = { cac_energy_maint = 0.00341 }               # 3.25 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^2 = cac_energy_maint }
	set_temp_variable = { cac_energy_maint = nuclear_reactors }
	multiply_temp_variable = { cac_energy_maint = 0.00945 }               # 9.00 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^2 = cac_energy_maint }
	set_temp_variable = { cac_energy_maint = renewable_energy_infra }
	multiply_temp_variable = { cac_energy_maint = 0.00893 }               # 8.50 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^2 = cac_energy_maint }
	multiply_variable = { CAC_companies_maintenance^2 = -1 }
	set_temp_variable = { cac_workforce_cost = fuel_silos }
	multiply_temp_variable = { cac_workforce_cost = 0.010 }               # workers_per_fuel_silo
	set_temp_variable = { cac_energy_workforce = number_of_fossil_pps }
	multiply_temp_variable = { cac_energy_workforce = 0.075 }             # workers_per_fossil_powerplant
	multiply_temp_variable = { cac_energy_workforce = fossil_powerplant_manpower_fulfillment }
	add_to_temp_variable = { cac_workforce_cost = cac_energy_workforce }
	set_temp_variable = { cac_energy_workforce = nuclear_reactors }
	multiply_temp_variable = { cac_energy_workforce = 0.100 }             # workers_per_nuclear_reactor
	multiply_temp_variable = { cac_energy_workforce = nuclear_reactor_manpower_fulfillment }
	add_to_temp_variable = { cac_workforce_cost = cac_energy_workforce }
	set_temp_variable = { cac_energy_workforce = renewable_energy_infra }
	multiply_temp_variable = { cac_energy_workforce = 0.030 }             # workers_per_renewable_energy
	multiply_temp_variable = { cac_energy_workforce = renewable_energy_infra_manpower_fulfillment }
	add_to_temp_variable = { cac_workforce_cost = cac_energy_workforce }
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	set_variable = { CAC_companies_workforce^2 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^2 = -1 }
	set_variable = { CAC_companies_sector_buildings^2 = fuel_silos }
	add_to_variable = { CAC_companies_sector_buildings^2 = number_of_fossil_pps }
	add_to_variable = { CAC_companies_sector_buildings^2 = nuclear_reactors }
	add_to_variable = { CAC_companies_sector_buildings^2 = renewable_energy_infra }

	# Slot 3 ‚Äî Defense (military factories + dockyards)
	set_variable = { CAC_companies_maintenance^3 = military_fac }
	add_to_variable = { CAC_companies_maintenance^3 = dockyard_fac }
	multiply_variable = { CAC_companies_maintenance^3 = 0.007875 }        # 7.50 * 0.15 * 0.007
	multiply_variable = { CAC_companies_maintenance^3 = -1 }
	set_temp_variable = { cac_workforce_cost = military_fac }
	add_to_temp_variable = { cac_workforce_cost = dockyard_fac }
	multiply_temp_variable = { cac_workforce_cost = 0.025 }               # workers_per_mil_fac / workers_per_dockyard
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	set_variable = { CAC_companies_workforce^3 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^3 = -1 }
	set_variable = { CAC_companies_sector_buildings^3 = military_fac }
	add_to_variable = { CAC_companies_sector_buildings^3 = dockyard_fac }

	# Slot 4 ‚Äî Technology & Communications (microchip plants + composite plants + internet stations)
	set_variable = { CAC_companies_maintenance^4 = microchip_plant_total }
	multiply_variable = { CAC_companies_maintenance^4 = 0.01575 }         # 15.00 * 0.15 * 0.007
	set_temp_variable = { cac_composite_maintenance = composite_plant_total }
	multiply_temp_variable = { cac_composite_maintenance = 0.01365 }      # 13.00 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^4 = cac_composite_maintenance }
	set_temp_variable = { cac_internet_maintenance = internet_station_total }
	multiply_temp_variable = { cac_internet_maintenance = 0.00315 }       # 3.00 * 0.15 * 0.007
	add_to_variable = { CAC_companies_maintenance^4 = cac_internet_maintenance }
	multiply_variable = { CAC_companies_maintenance^4 = -1 }
	set_temp_variable = { cac_workforce_cost = microchip_plant_total }
	multiply_temp_variable = { cac_workforce_cost = 0.050 }               # workers_per_microchip_plant
	set_temp_variable = { cac_composite_workforce = composite_plant_total }
	multiply_temp_variable = { cac_composite_workforce = 0.050 }          # workers_per_composite_plant
	add_to_temp_variable = { cac_workforce_cost = cac_composite_workforce }
	set_temp_variable = { cac_internet_workforce = internet_station_total }
	multiply_temp_variable = { cac_internet_workforce = 0.030 }           # workers_per_internet_station
	multiply_temp_variable = { cac_internet_workforce = internet_station_manpower_fulfillment }
	add_to_temp_variable = { cac_workforce_cost = cac_internet_workforce }
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	set_variable = { CAC_companies_workforce^4 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^4 = -1 }
	set_variable = { CAC_companies_sector_buildings^4 = microchip_plant_total }
	add_to_variable = { CAC_companies_sector_buildings^4 = composite_plant_total }
	add_to_variable = { CAC_companies_sector_buildings^4 = internet_station_total }

	# Slot 5 ‚Äî Construction & Infrastructure (civilian factories)
	set_variable = { CAC_companies_maintenance^5 = industrial_complex_total }
	multiply_variable = { CAC_companies_maintenance^5 = 0.007875 }        # 7.50 * 0.15 * 0.007
	multiply_variable = { CAC_companies_maintenance^5 = -1 }
	set_temp_variable = { cac_workforce_cost = industrial_complex_total }
	multiply_temp_variable = { cac_workforce_cost = 0.275 }               # workers_per_civ_fac
	multiply_temp_variable = { cac_workforce_cost = cac_gdpc_scaled }
	set_variable = { CAC_companies_workforce^5 = cac_workforce_cost }
	multiply_variable = { CAC_companies_workforce^5 = -1 }
	set_variable = { CAC_companies_sector_buildings^5 = industrial_complex_total }

	# Slot 6 ‚Äî Healthcare & Pharma (scales with state_hospital_need)
	#TODO: add modifier so that increased health requires increased maintenance
	set_variable = { CAC_companies_maintenance^6 = state_hospital_need }
	multiply_variable = { CAC_companies_maintenance^6 = 0.175 }
	multiply_variable = { CAC_companies_maintenance^6 = -1 }
	set_variable = { CAC_companies_workforce^6 = state_hospital_need }
	multiply_variable = { CAC_companies_workforce^6 = cac_gdpc_scaled }
	multiply_variable = { CAC_companies_workforce^6 = -1 }
	set_variable = { CAC_companies_sector_buildings^6 = 0 }
}

# CAC_update_variables
# Purpose: Computes per-company taxes (needed for corporate_tax ‚Üí tax_gain) and, for human
# players only, refreshes income/revenue/display_expenses display arrays.
# Called from calculate_tax_gain weekly and from CAC_modify_* effects on player actions.
# Guarded by cac_skip_all_updates during startup initialization.
# Maintenance and workforce are pre-computed monthly (stored as negative) by
# CAC_update_building_stats and added directly into revenue/display_expenses here.
CAC_update_variables = {
	if = {
		limit = { NOT = { check_variable = { cac_skip_all_updates = 1 } } }

		# cac_gdpc_scaled needed for energy income (slot 2) ‚Äî computed for both AI and human
		set_temp_variable = { cac_gdpc_scaled = gdp_per_capita }
		multiply_temp_variable = { cac_gdpc_scaled = 0.007 }

		# Slot 0 ‚Äî Banking & Finance (offices)
		# Tax stored positive, used in display block, then negated for corporate_tax subtraction trick.
		set_variable = { CAC_companies_taxes^0 = office_tax }
		if = { limit = { is_ai = no }
			set_variable = { CAC_companies_income^0 = office_gross_income }
			set_variable = { CAC_companies_revenue^0 = CAC_companies_income^0 }
			subtract_from_variable = { CAC_companies_revenue^0 = CAC_companies_taxes^0 }
			add_to_variable = { CAC_companies_revenue^0 = CAC_companies_additional_income^0 }
			subtract_from_variable = { CAC_companies_revenue^0 = CAC_companies_additional_expenses^0 }
			add_to_variable = { CAC_companies_revenue^0 = CAC_companies_maintenance^0 }
			add_to_variable = { CAC_companies_revenue^0 = CAC_companies_workforce^0 }
			set_variable = { CAC_companies_display_expenses^0 = 0 }
			subtract_from_variable = { CAC_companies_display_expenses^0 = CAC_companies_taxes^0 }
			subtract_from_variable = { CAC_companies_display_expenses^0 = CAC_companies_additional_expenses^0 }
			add_to_variable = { CAC_companies_display_expenses^0 = CAC_companies_maintenance^0 }
			add_to_variable = { CAC_companies_display_expenses^0 = CAC_companies_workforce^0 }
		}
		multiply_variable = { CAC_companies_taxes^0 = -1 }

		# Slot 1 ‚Äî Mining & Resources (resource exports + synthetic refineries + agriculture)
		set_variable = { CAC_companies_taxes^1 = synthetic_refinery_tax }
		add_to_variable = { CAC_companies_taxes^1 = agriculture_district_tax }
		if = { limit = { is_ai = no }
			set_variable = { CAC_companies_income^1 = CAC_resource_sale_cic_adjusted }
			add_to_variable = { CAC_companies_income^1 = synthetic_gross_income }
			add_to_variable = { CAC_companies_income^1 = agriculture_district_gross_income }
			set_variable = { CAC_companies_revenue^1 = CAC_companies_income^1 }
			subtract_from_variable = { CAC_companies_revenue^1 = CAC_companies_taxes^1 }
			add_to_variable = { CAC_companies_revenue^1 = CAC_companies_additional_income^1 }
			subtract_from_variable = { CAC_companies_revenue^1 = CAC_companies_additional_expenses^1 }
			add_to_variable = { CAC_companies_revenue^1 = CAC_companies_maintenance^1 }
			add_to_variable = { CAC_companies_revenue^1 = CAC_companies_workforce^1 }
			set_variable = { CAC_companies_display_expenses^1 = 0 }
			subtract_from_variable = { CAC_companies_display_expenses^1 = CAC_companies_taxes^1 }
			subtract_from_variable = { CAC_companies_display_expenses^1 = CAC_companies_additional_expenses^1 }
			add_to_variable = { CAC_companies_display_expenses^1 = CAC_companies_maintenance^1 }
			add_to_variable = { CAC_companies_display_expenses^1 = CAC_companies_workforce^1 }
		}
		multiply_variable = { CAC_companies_taxes^1 = -1 }

		# Slot 2 ‚Äî Energy & Utilities (fuel silos + fossil powerplants + nuclear reactors + renewable energy)
		# Income computed in temps; tax stored directly from temp (avoids reading persistent income^2).
		set_temp_variable = { cac_energy_income = fuel_silos }
		multiply_temp_variable = { cac_energy_income = 0.05 }            # income_rate_fuel_silo
		set_temp_variable = { cac_energy_temp = number_of_fossil_pps }
		multiply_temp_variable = { cac_energy_temp = 0.35 }              # income_rate_fossil_powerplant
		add_to_temp_variable = { cac_energy_income = cac_energy_temp }
		set_temp_variable = { cac_energy_temp = nuclear_reactors }
		multiply_temp_variable = { cac_energy_temp = 0.60 }              # income_rate_nuclear_reactor
		add_to_temp_variable = { cac_energy_income = cac_energy_temp }
		set_temp_variable = { cac_energy_temp = renewable_energy_infra }
		multiply_temp_variable = { cac_energy_temp = 0.20 }              # income_rate_renewable_energy
		add_to_temp_variable = { cac_energy_income = cac_energy_temp }
		multiply_temp_variable = { cac_energy_income = cac_gdpc_scaled }
		set_temp_variable = { energy_consumption_mod = 1 }
		add_to_temp_variable = { energy_consumption_mod = modifier@energy_sector_consumption_multiplier_modifier }
		multiply_temp_variable = { cac_energy_income = energy_consumption_mod }
		set_variable = { CAC_companies_taxes^2 = cac_energy_income }
		multiply_variable = { CAC_companies_taxes^2 = corporate_tax_rate }
		multiply_variable = { CAC_companies_taxes^2 = 0.01 }
		if = { limit = { is_ai = no }
			set_variable = { CAC_companies_income^2 = cac_energy_income }
			set_variable = { CAC_companies_revenue^2 = CAC_companies_income^2 }
			subtract_from_variable = { CAC_companies_revenue^2 = CAC_companies_taxes^2 }
			add_to_variable = { CAC_companies_revenue^2 = CAC_companies_additional_income^2 }
			subtract_from_variable = { CAC_companies_revenue^2 = CAC_companies_additional_expenses^2 }
			add_to_variable = { CAC_companies_revenue^2 = CAC_companies_maintenance^2 }
			add_to_variable = { CAC_companies_revenue^2 = CAC_companies_workforce^2 }
			set_variable = { CAC_companies_display_expenses^2 = 0 }
			subtract_from_variable = { CAC_companies_display_expenses^2 = CAC_companies_taxes^2 }
			subtract_from_variable = { CAC_companies_display_expenses^2 = CAC_companies_additional_expenses^2 }
			add_to_variable = { CAC_companies_display_expenses^2 = CAC_companies_maintenance^2 }
			add_to_variable = { CAC_companies_display_expenses^2 = CAC_companies_workforce^2 }
		}
		multiply_variable = { CAC_companies_taxes^2 = -1 }

		# Slot 3 ‚Äî Defense (military factories + dockyards)
		set_variable = { CAC_companies_taxes^3 = military_fac_tax }
		add_to_variable = { CAC_companies_taxes^3 = dockyard_tax }
