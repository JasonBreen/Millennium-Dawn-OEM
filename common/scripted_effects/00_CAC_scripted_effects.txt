# CAC_add_company
# Required Input: cac_selected_company (int)
# Required Input: cac_company_type (int)

# Optional Inputs:
# cac_company_size (int)
# cac_company_value (float)
# cac_company_money (float)
# cac_company_factories (int)
# cac_company_income_per_factory (float)
# cac_company_expenses_per_factory (float)
# cac_company_additional_income (float)
# cac_company_additional_expenses (float)

# Purpose: Add the cac_selected_company to companys list
CAC_add_company = {
    if = {
        limit = {
            check_variable = { CAC_companies^num < 11  }
        }

        custom_effect_tooltip = CAC_add_company_tt

        set_temp_variable = { company_slot_temp = CAC_companies^num }

        add_to_array = {
		    array = CAC_companies
			value = cac_selected_company
			index = company_slot_temp
		}

        add_to_array = {
		    array = CAC_companies_type
			value = cac_company_type
			index = company_slot_temp
		}

        add_to_array = {
            array = CAC_companies_size
            value = cac_company_size
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_value
            value = cac_company_value
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_money
            value = 0
            index = company_slot_temp
        }

        set_temp_variable = { cac_skip_update = 1 }

        set_temp_variable = { cac_money_change = 5 }
        set_temp_variable = { cac_slot = company_slot_temp }
        CAC_modify_money_effect = yes

        add_to_array = {
            array = CAC_companies_revenue
            value = 0
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_factories
            value = 0
            index = company_slot_temp
        }

        if = {
            limit = { check_variable = { cac_company_factories = 0 } }

            set_temp_variable = { cac_company_factories = 1 }
        }

        set_temp_variable = { cac_factories_change = cac_company_factories }
        set_temp_variable = { cac_slot = company_slot_temp }
        CAC_modify_factories_effect = yes

        add_to_array = {
            array = CAC_companies_income_per_factory
            value = 0
            index = company_slot_temp
        }

        if = {
            limit = { check_variable = { cac_company_income_per_factory = 0 } }

            set_temp_variable = { cac_company_income_per_factory = 0.5 }
        }

        set_temp_variable = { cac_income_per_factory_change = cac_company_income_per_factory }
        set_temp_variable = { cac_slot = company_slot_temp }
        CAC_modify_income_per_factory_effect = yes

        add_to_array = {
            array = CAC_companies_expenses_per_factory
            value = 0
            index = company_slot_temp
        }

        if = {
            limit = { check_variable = { cac_company_expenses_per_factory = 0 } }

            set_temp_variable = { cac_company_expenses_per_factory = -1.35 } # DEBUG 0.35
        }

        set_temp_variable = { cac_expenses_per_factory_change = cac_company_expenses_per_factory }
        set_temp_variable = { cac_slot = company_slot_temp }
        CAC_modify_expenses_per_factory_effect = yes

        if = {
            limit = {
                NOT = { check_variable = { cac_company_additional_income = 0 } }
            }

            add_to_array = {
                array = CAC_companies_additional_income
                value = cac_company_additional_income
                index = company_slot_temp
            }
        }
        else = {
            add_to_array = {
                array = CAC_companies_additional_income
                value = 0
                index = company_slot_temp
            }
        }

        if = {
            limit = {
                NOT = { check_variable = { cac_company_additional_expenses = 0 } }
            }

            add_to_array = {
                array = CAC_companies_additional_expenses
                value = cac_company_additional_expenses
                index = company_slot_temp
            }
        }
        else = {
            add_to_array = {
                array = CAC_companies_additional_expenses
                value = 0
                index = company_slot_temp
            }
        }

        add_to_array = {
			array = CAC_companies_income
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
			array = CAC_companies_expenses
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
			array = CAC_companies_taxes
			value = 0.0
			index = company_slot_temp
		}
        add_to_array = {
            array = CAC_companies_display_expenses
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_maintenance
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_workforce
            value = 0.0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_sector_buildings
            value = 0
            index = company_slot_temp
        }

        # company modifiers
        add_to_array = {
            array = CAC_companies_income_modifier
            value = 0
            index = company_slot_temp
        }
        add_to_array = {
            array = CAC_companies_expenses_modifier
            value = 0
            index = company_slot_temp
        }

        add_to_array = {
            array = CAC_companies_modifier_from_size
            value = 0
            index = company_slot_temp
        }

        set_temp_variable = { cac_skip_update = 0 }
        CAC_update_variables = yes

        # upgrades
        if = {
            limit = { check_variable = { CAC_companies_upgrades_2d^num = 0 } }

            for_loop_effect = {
                start = 0
                end = 11
                value = company_index

                for_loop_effect = {
                    start = 0
                    end = 30
                    value = upgrade_id

                    set_temp_variable = { index = company_index }
                    multiply_temp_variable = { index = 30 }
                    add_to_temp_variable = { index = upgrade_id }

                    add_to_array = {
                        array = CAC_companies_upgrades_2d
                        value = 3
                        index = index
                    }
                }
            }
        }

        if = {
            limit = { check_variable = { cac_company_type = 0 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 1 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 2 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 3 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 4 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 5 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
        else_if = {
            limit = { check_variable = { cac_company_type = 6 } }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 0 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }

            set_temp_variable = { index = company_slot_temp }
            multiply_temp_variable = { index = 30 }
            add_to_temp_variable = { index = 11 }

            set_variable = { CAC_companies_upgrades_2d^index = 2 }
        }
    }
}

# CAC_modify_money_effect
# Required Input: cac_money_change (float)
# Required Input: cac_slot (int)
# Purpose: Change company money on slot cac_slot by cac_money_change
CAC_modify_money_effect = {
    if = {
        limit = {
            AND = {
                check_variable = { cac_slot > -1 }
                check_variable = { cac_slot < CAC_companies^num }
            }
        }

        custom_effect_tooltip = CAC_modify_money_tt

        add_to_variable = { CAC_companies_money^cac_slot = cac_money_change }
        add_to_variable = { CAC_companies_value^cac_slot = cac_money_change }

        CAC_update_variables = yes
    }
}

# CAC_modify_upgrade_status_effect
# Required Input: cac_slot (int)

# Required Input: cac_new_status (int) 1 - completed 2 - available 3 - blocked
# Purpose: Change company upgrade status on slot cac_slot by cac_new_status
CAC_modify_upgrade_status_effect = {
    if = {
        limit = {
            AND = {
                check_variable = { cac_slot > -1 }
                check_variable = { cac_slot < CAC_companies^num }
            }
        }

        set_temp_variable = { index = cac_slot }
		multiply_temp_variable = { index = 30 }
		add_to_temp_variable = { index = cac_upgrade }

		set_variable = { CAC_companies_upgrades_2d^index = cac_new_status }

        if = {
            limit = { check_variable = { cac_new_status = 1 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_1_tt
        }
        else_if = {
            limit = { check_variable = { cac_new_status = 2 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_2_tt
        }
        else_if = {
            limit = { check_variable = { cac_new_status = 3 } }
            #custom_effect_tooltip = CAC_modify_upgrade_status_3_tt
        }
    }
}

# CAC_complete_upgrade_effect
# Required Input: cac_base_money_change (int)
# Required Input: cac_slot (int)
# Purpose: Remove money from company on upgrade
CAC_complete_upgrade_effect = {
    set_temp_variable = { cac_money_change_modifier = 1 }
    add_to_temp_variable = { cac_money_change_modifier = CAC_companies_modifier_from_size^cac_slot }

    multiply_temp_variable = { cac_base_money_change = cac_money_change_modifier }

    add_to_variable = { CAC_companies_money^cac_slot = cac_base_money_change }
    add_to_variable = { CAC_companies_value^cac_slot = cac_base_money_change }

    CAC_update_variables = yes
}

# CAC_calculate_sector_pool
# Purpose: Compute the national corporate income pool from real building counts
CAC_calculate_sector_pool = {
    # Shared scaling: control_factor × overall_productivity (normalized) × weekly factor
    # Matches the structure of calculate_tax_gain's unified_building_multiplier
    set_temp_variable = { cac_pool_scale = overall_productivity }
    multiply_temp_variable = { cac_pool_scale = 0.001 }
    multiply_temp_variable = { cac_pool_scale = 0.007 }
    multiply_temp_variable = { cac_pool_scale = @cac_control_factor }

    # Office Parks — base 5 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = office_park_total }
    multiply_temp_variable = { temp_pool = offices_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 5.0 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    set_variable = { CAC_sector_pool = temp_pool }

    # Synthetic Refineries — base 3 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = synthetic_refinery_total }
    multiply_temp_variable = { temp_pool = synthetic_refinery_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 3.0 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Agriculture Districts — base 2.6 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = agriculture_district_total }
    multiply_temp_variable = { temp_pool = agriculture_district_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 2.6 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Renewable Energy — no tax_gain equivalent; keep relative weight 1.0, no fulfillment variable
    set_temp_variable = { temp_pool = renewable_energy_infra }
    multiply_temp_variable = { temp_pool = 1.0 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Nuclear Reactors — no tax_gain equivalent; keep relative weight 1.5, no fulfillment variable
    set_temp_variable = { temp_pool = nuclear_reactors }
    multiply_temp_variable = { temp_pool = 1.5 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Fuel Silos — no tax_gain equivalent; keep relative weight 0.5, no fulfillment variable
    set_temp_variable = { temp_pool = fuel_silos }
    multiply_temp_variable = { temp_pool = 0.5 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Military Factories — base 0.2 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = military_fac }
    multiply_temp_variable = { temp_pool = military_factories_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 0.2 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Microchip Plants — base 4 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = microchip_plant_total }
    multiply_temp_variable = { temp_pool = microchip_plants_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 4.0 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Industrial Complexes (CIC) — base 2.5 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = industrial_complex_total }
    multiply_temp_variable = { temp_pool = civilian_factories_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 2.5 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Composite Plants — base 3.5 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = composite_plant_total }
    multiply_temp_variable = { temp_pool = composite_plants_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 3.5 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    # Naval Dockyards — base 0.2 (matches calculate_tax_gain)
    set_temp_variable = { temp_pool = dockyard_fac }
    multiply_temp_variable = { temp_pool = naval_factories_manpower_fulfillment }
    multiply_temp_variable = { temp_pool = 0.2 }
    multiply_temp_variable = { temp_pool = cac_pool_scale }
    add_to_variable = { CAC_sector_pool = temp_pool }

    clamp_variable = { var = CAC_sector_pool min = 1 }
}

# CAC_calculate_sector_weights
# Purpose: Compute per-sector weight and total weight for income distribution
CAC_calculate_sector_weights = {
    # Sector 0: Banking & Finance
    set_variable = { CAC_sector_weight_0 = office_park_total }
    multiply_variable = { CAC_sector_weight_0 = 5.0 }
    add_to_variable = { CAC_sector_weight_0 = 2.0 }

    # Sector 1: Mining & Resources
    set_variable = { CAC_sector_weight_1 = synthetic_refinery_total }

    set_temp_variable = { temp_w = fuel_silos }

    add_to_variable = { CAC_sector_weight_1 = temp_w }
    set_temp_variable = { temp_w = agriculture_district_total }
    multiply_temp_variable = { temp_w = 2.5 }
    add_to_variable = { CAC_sector_weight_1 = temp_w }
    add_to_variable = { CAC_sector_weight_1 = 1.0 }

    # Sector 2: Energy & Utilities
    set_variable = { CAC_sector_weight_2 = renewable_energy_infra }
    multiply_variable = { CAC_sector_weight_2 = 2.5 }
    set_temp_variable = { temp_w = nuclear_reactors }
    multiply_temp_variable = { temp_w = 3.0 }
    add_to_variable = { CAC_sector_weight_2 = temp_w }
    add_to_variable = { CAC_sector_weight_2 = 0.5 }

    # Sector 3: Defense
    set_variable = { CAC_sector_weight_3 = military_fac }
    multiply_variable = { CAC_sector_weight_3 = 2.0 }
    set_temp_variable = { temp_w = dockyard_fac }
    multiply_temp_variable = { temp_w = 2.0 }
    add_to_variable = { CAC_sector_weight_3 = temp_w }
    add_to_variable = { CAC_sector_weight_3 = 0.5 }

    # Sector 4: Media & Telecommunications
    set_variable = { CAC_sector_weight_4 = microchip_plant_total }
    multiply_variable = { CAC_sector_weight_4 = 4.0 }
    set_temp_variable = { temp_w = composite_plant_total }
    multiply_temp_variable = { temp_w = 1.75 }
    add_to_variable = { CAC_sector_weight_4 = temp_w }
    add_to_variable = { CAC_sector_weight_4 = 1.5 }

    # Sector 5: Construction & Infrastructure
    set_variable = { CAC_sector_weight_5 = industrial_complex_total }
    multiply_variable = { CAC_sector_weight_5 = 1.5 }
    add_to_variable = { CAC_sector_weight_5 = 1.0 }

    # Sector 6: Pharmaceuticals & Healthcare (GDP-derived)
    set_variable = { CAC_sector_weight_6 = gdp_per_capita }
    multiply_variable = { CAC_sector_weight_6 = 0.002 }
    clamp_variable = { var = CAC_sector_weight_6 min = 0.5 }

    # Sum all weights (sector 1 excluded — draws from resource_sale_rate, not the pool)
    set_variable = { CAC_total_sector_weight = CAC_sector_weight_0 }
    add_to_variable = { CAC_total_sector_weight = CAC_sector_weight_2 }
    add_to_variable = { CAC_total_sector_weight = CAC_sector_weight_3 }
    add_to_variable = { CAC_total_sector_weight = CAC_sector_weight_4 }
    add_to_variable = { CAC_total_sector_weight = CAC_sector_weight_5 }
    add_to_variable = { CAC_total_sector_weight = CAC_sector_weight_6 }
    clamp_variable = { var = CAC_total_sector_weight min = 1 }
}

CAC_update_variables = {
    if = {
        limit = {
            OR = {
                check_variable = { cac_skip_update = 1 }
                check_variable = { cac_skip_all_updates = 1 }
            }
        }
    }
    else_if = {
        limit = {
            check_variable = { CAC_companies^num > 0 }
        }


        set_variable = { CAC_mining_tax_income = 0 }
        set_temp_variable = { cac_pool_ratio = CAC_sector_pool }
        clamp_variable = { var = CAC_total_sector_weight min = 1 }
        divide_temp_variable = { cac_pool_ratio = CAC_total_sector_weight }
        set_temp_variable = { cac_pool_scale = overall_productivity }
        multiply_temp_variable = { cac_pool_scale = 0.001 }
        multiply_temp_variable = { cac_pool_scale = 0.007 }
        multiply_temp_variable = { cac_pool_scale = @cac_control_factor }

        set_temp_variable = { cac_global_income_mod = 1 }
        add_to_temp_variable = { cac_global_income_mod = modifier@companies_income_modifier }

        set_temp_variable = { cac_global_expense_mod = 1 }

        set_temp_variable = { cac_tax_base_mult = corporate_tax_rate }
