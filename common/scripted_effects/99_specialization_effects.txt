# ============================================================================
# COUNTRY SPECIALIZATION — Building Augmentation System
# ----------------------------------------------------------------------------
# Global mechanic. Each building type can be activated and augmented with raw
# resources; each resource gets a 0-5 contribution LEVEL set with +/- arrows.
# A building's strength = the sum of its resource levels. Augmentation draws GW
# (electricity), workers (employment) and diverts the chosen resources from
# export, all wired into the live MD economy.
#
# State: spec_active_<b> (0/1)  +  spec_lvl_<b>_<token> (0-5)
# Building keys: civ mil dock office agri refinery chip comp
# ============================================================================

spec_set_tunables = {
	# Re-applied on every startup/load (not flag-gated) so tuning takes effect on existing saves.
	set_variable = { global.spec_per_factory_units = 0.15 }      # resource units consumed per factory, per level (TUNE)
	set_variable = { global.spec_building_cap = 0.15 }           # max fraction of a resource ONE building can divert (when maxed)
	set_variable = { global.spec_resource_cap = 0.5 }            # hard cap: max fraction of a resource diverted across ALL buildings
	set_variable = { global.spec_buff_base = 0.01 }              # buff per level x rule-weight (TUNE)
	set_variable = { global.spec_power_flat = 0.02 }             # flat power-use bump per building with active resources (TUNE)
}
