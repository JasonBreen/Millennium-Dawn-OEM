#   Social: Xi [59,60,61,62,63]
#   NSC:    Xi [64,65,66,67,68]
CHI_assign_all_secretaries = {
	# Slot 1
	if = {
		limit = { check_variable = { CHI_ic_slot_1 > 0 } }
		set_temp_variable = { CHI_ic_assign_slot = 1 }
		if = { limit = { check_variable = { CHI_ic_slot_1 = 1 } } CHI_assign_org_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 2 } } CHI_assign_pub_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 3 } } CHI_assign_ufwd_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 4 } } CHI_assign_intl_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 5 } } CHI_assign_social_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 6 } } CHI_assign_cplac_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_1 = 7 } } CHI_assign_nsc_secretaries = yes }
	}
	# Slot 2
	if = {
		limit = { check_variable = { CHI_ic_slot_2 > 0 } }
		set_temp_variable = { CHI_ic_assign_slot = 2 }
		if = { limit = { check_variable = { CHI_ic_slot_2 = 1 } } CHI_assign_org_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 2 } } CHI_assign_pub_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 3 } } CHI_assign_ufwd_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 4 } } CHI_assign_intl_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 5 } } CHI_assign_social_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 6 } } CHI_assign_cplac_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_2 = 7 } } CHI_assign_nsc_secretaries = yes }
	}
	# Slot 3
	if = {
		limit = { check_variable = { CHI_ic_slot_3 > 0 } }
		set_temp_variable = { CHI_ic_assign_slot = 3 }
		if = { limit = { check_variable = { CHI_ic_slot_3 = 1 } } CHI_assign_org_secretaries = yes }
		else_if = { limit = { check_variable = { CHI_ic_slot_3 = 2 } } CHI_assign_pub_secretaries = yes }

# Helper — copies temp picks to the correct slot variables
CHI_ic_copy_picks_to_slot = {
	if = {
		limit = { check_variable = { CHI_ic_assign_slot = 1 } }
		set_variable = { CHI_ic_s1_sec1 = CHI_ic_pick1 }
		set_variable = { CHI_ic_s1_sec2 = CHI_ic_pick2 }
		set_variable = { CHI_ic_s1_sec3 = CHI_ic_pick3 }
	}
	else_if = {
		limit = { check_variable = { CHI_ic_assign_slot = 2 } }
		set_variable = { CHI_ic_s2_sec1 = CHI_ic_pick1 }
		set_variable = { CHI_ic_s2_sec2 = CHI_ic_pick2 }
		set_variable = { CHI_ic_s2_sec3 = CHI_ic_pick3 }
	}
	else_if = {
		limit = { check_variable = { CHI_ic_assign_slot = 3 } }
		set_variable = { CHI_ic_s3_sec1 = CHI_ic_pick1 }
		set_variable = { CHI_ic_s3_sec2 = CHI_ic_pick2 }
		set_variable = { CHI_ic_s3_sec3 = CHI_ic_pick3 }
	}
}
