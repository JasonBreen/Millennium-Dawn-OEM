### BRI — target state selection wrapper ###
# Caller sets set_temp_variable = { project_type = N } on the recipient scope before invoking.
# 1=port  2=railway  3=pipeline  4=telecom  5=SEZ  6=mine  7=powerplant  8=prestige
# Picks one owned-controlled state matching the type, applies the matching state dynamic modifier
# and a state flag, and increments the corresponding CHI counter.
BRI_select_target_state = {
	if = {
		limit = { check_variable = { project_type = 1 } }
		random_owned_controlled_state = {
			limit = { is_coastal = yes }
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
			set_state_flag = bri_port_built
		}
		CHI = { add_to_variable = { CHI_bri_ports_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 2 } }
		random_owned_controlled_state = {
			limit = {
				any_neighbor_state = { CONTROLLER = { original_tag = CHI } }
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
			set_state_flag = bri_railway_built
		}
		CHI = { add_to_variable = { CHI_bri_railways_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 3 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					has_resources_amount = { resource = oil amount > 0 }
					has_resources_amount = { resource = steel amount > 0 }
					has_resources_amount = { resource = chromium amount > 0 }
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
			set_state_flag = bri_pipeline_built
		}
		CHI = { add_to_variable = { CHI_bri_pipelines_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 4 } }
		random_owned_controlled_state = {
			limit = { industrial_complex > 0 }
			add_dynamic_modifier = { modifier = CHI_bri_state_information }
			set_state_flag = bri_telecom_built
		}
		CHI = { add_to_variable = { CHI_bri_telecom_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 5 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					is_coastal = yes
					infrastructure > 3
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
			set_state_flag = bri_sez_built
		}
		CHI = { add_to_variable = { CHI_bri_industrial_zones = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 6 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					has_resources_amount = { resource = steel amount > 0 }
					has_resources_amount = { resource = chromium amount > 0 }
					has_resources_amount = { resource = aluminium amount > 0 }
					has_resources_amount = { resource = tungsten amount > 0 }
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
			set_state_flag = bri_mine_built
		}
		CHI = { add_to_variable = { CHI_bri_mines_secured = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 7 } }
		random_owned_controlled_state = {
			limit = { industrial_complex > 0 }
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
			set_state_flag = bri_powerplant_built
		}
		CHI = { add_to_variable = { CHI_bri_powerplants_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 8 } }
		random_owned_controlled_state = {
			limit = { is_capital = yes }
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
			set_state_flag = bri_prestige_built
		}
		CHI = { add_to_variable = { CHI_bri_prestige_projects = 1 } }
	}
}

### BRI — register recipient as partner and recalc tier ###
# Called on the recipient country scope after BRI_select_target_state.
# Increments per-country project counter, bumps CHI presence/completion counters,
# and swaps the recipient onto the correct partner tier dynamic modifier.
BRI_register_and_recalc_partner = {
	add_to_variable = { bri_project_states = 1 }
	CHI = { add_to_variable = { CHI_bri_global_presence = 5 } }
	CHI = { add_to_variable = { CHI_bri_completed_projects = 1 } }
	set_country_flag = bri_partner

	if = {
		limit = {
			check_variable = { bri_project_states > 4 }
		}
		if = {
			limit = { has_dynamic_modifier = { modifier = bri_partner_tier_2 } }
			remove_dynamic_modifier = { modifier = bri_partner_tier_2 }
		}
		if = {
			limit = { has_dynamic_modifier = { modifier = bri_partner_tier_1 } }
			remove_dynamic_modifier = { modifier = bri_partner_tier_1 }
		}
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = bri_partner_tier_3 } } }
			add_dynamic_modifier = { modifier = bri_partner_tier_3 }
		}
	}
	else_if = {
		limit = {
			check_variable = { bri_project_states > 2 }
		}
		if = {
			limit = { has_dynamic_modifier = { modifier = bri_partner_tier_1 } }
			remove_dynamic_modifier = { modifier = bri_partner_tier_1 }
		}
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = bri_partner_tier_2 } } }
			add_dynamic_modifier = { modifier = bri_partner_tier_2 }
		}
	}
	else = {
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = bri_partner_tier_1 } } }
			add_dynamic_modifier = { modifier = bri_partner_tier_1 }
		}
	}
}

### BRI — resolve state modifier change (decay or asset seizure) ###
# State scope at call time. Caller sets set_temp_variable = { resolve_mode = X }:
#   resolve_mode = 0 -> decay (strip the regular modifier; nothing replaces it)
#   resolve_mode = 1 -> asset seizure (strip regular, attach the chinese_controlled variant)
BRI_resolve_state_change = {
	if = {
		limit = { has_dynamic_modifier = { modifier = CHI_bri_state_infrastructure } }
		remove_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
		if = {
			limit = { check_variable = { resolve_mode = 1 } }
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure_controlled }
		}
	}
	else_if = {
		limit = { has_dynamic_modifier = { modifier = CHI_bri_state_industrial } }
		remove_dynamic_modifier = { modifier = CHI_bri_state_industrial }
		if = {
			limit = { check_variable = { resolve_mode = 1 } }
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial_controlled }
		}
	}
	else_if = {
		limit = { has_dynamic_modifier = { modifier = CHI_bri_state_information } }
		remove_dynamic_modifier = { modifier = CHI_bri_state_information }
		if = {
			limit = { check_variable = { resolve_mode = 1 } }
			add_dynamic_modifier = { modifier = CHI_bri_state_information_controlled }
		}
	}
}

### Uyghur Threat: monthly tick ###
# Called from on_monthly when CHI controls state 592.
# Drifts uighur_threat based on state 592 resistance, then rolls for uprising events.
CHI_uighur_monthly_tick = {
	# Guard: skip if ETK controls state 592
	if = {
		limit = {
			NOT = { 592 = { is_controlled_by = ETK } }
		}

		### Drift step: resistance drives uighur_threat up or down ###
		if = {
			limit = { 592 = { resistance > 60 } }
			set_temp_variable = { modify_uighur_threat = 2 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = { 592 = { resistance > 39 } }
			set_temp_variable = { modify_uighur_threat = 1 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = { 592 = { NOT = { resistance > 14 } } }
			set_temp_variable = { modify_uighur_threat = -1 }
			CHI_modify_uighur_threat = yes
		}

		### Secondary push from sinicization ###
		if = {
			limit = {
				check_variable = {
					var = ETK_sinicization_var
					value = 60
					compare = greater_than
				}
			}
			set_temp_variable = { modify_uighur_threat = 1 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = {
				check_variable = {
					var = ETK_sinicization_var
					value = 20
					compare = less_than
				}
			}
			set_temp_variable = { modify_uighur_threat = -1 }
			CHI_modify_uighur_threat = yes
		}

		### Event roll: gated by cooldown flag ###
		if = {
			limit = {
				NOT = { has_country_flag = CHI_uighur_event_cd }
			}

			# Calm tier: uighur_threat 0..29
			if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 30
						compare = less_than
					}
				}
				random_list = {
					92 = { }
					8 = {
						country_event = { id = chi_uighur.5 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
				}
			}
			# Tense tier: uighur_threat 30..59
			else_if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 60
						compare = less_than
					}
				}
				random_list = {
					60 = { }
					30 = {
						country_event = { id = chi_uighur.1 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
					10 = {
						country_event = { id = chi_uighur.2 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
				}
			}
			# Volatile tier: uighur_threat 60..84
			else_if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 85
						compare = less_than
					}
				}
				random_list = {
					35 = { }
					25 = {
						country_event = { id = chi_uighur.1 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
					30 = {
						country_event = { id = chi_uighur.2 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
					10 = {
						country_event = { id = chi_uighur.3 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
				}
			}
			# Boiling tier: uighur_threat 85..100
			else = {
				random_list = {
					15 = { }
					15 = {
						country_event = { id = chi_uighur.1 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
					35 = {

					}
					35 = {
						country_event = { id = chi_uighur.3 }
						set_country_flag = { flag = CHI_uighur_event_cd days = 60 }
					}
				}

		}

# Swaps the CHI_semiconductor_backwardness idea down a tier (or removes it) once self-reliance
# crosses a threshold. Idempotent: each branch only acts on the tier actually held, and jumps
# straight to the correct tier so a large single gain (e.g. the leverage event) skips levels.
CHI_evaluate_semiconductor_boycott = {
	if = {
		limit = {
			has_idea = CHI_semiconductor_backwardness
			check_variable = { CHI_semiconductor_self_reliance > 24 }
		}
		if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 99 } }
			remove_ideas = CHI_semiconductor_backwardness
		}
		else_if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 74 } }
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness add_idea = CHI_semiconductor_backwardness_4 }
		}
		else_if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 49 } }
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness add_idea = CHI_semiconductor_backwardness_3 }
		}
		else = {
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness add_idea = CHI_semiconductor_backwardness_2 }
		}
	}
	else_if = {
		limit = {
			has_idea = CHI_semiconductor_backwardness_2
			check_variable = { CHI_semiconductor_self_reliance > 49 }
		}
		if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 99 } }
			remove_ideas = CHI_semiconductor_backwardness_2
		}
		else_if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 74 } }
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness_2 add_idea = CHI_semiconductor_backwardness_4 }
		}
		else = {
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness_2 add_idea = CHI_semiconductor_backwardness_3 }
		}
	}
	else_if = {
		limit = {
			has_idea = CHI_semiconductor_backwardness_3
			check_variable = { CHI_semiconductor_self_reliance > 74 }
		}
		if = {
			limit = { check_variable = { CHI_semiconductor_self_reliance > 99 } }
			remove_ideas = CHI_semiconductor_backwardness_3
		}
		else = {
			swap_ideas = { remove_idea = CHI_semiconductor_backwardness_3 add_idea = CHI_semiconductor_backwardness_4 }
		}
	}
	else_if = {
		limit = {
			has_idea = CHI_semiconductor_backwardness_4
			check_variable = { CHI_semiconductor_self_reliance > 99 }
		}
		remove_ideas = CHI_semiconductor_backwardness_4
	}
}
