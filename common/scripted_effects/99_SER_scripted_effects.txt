modify_voj_separ_support = {
	add_to_variable = { unions_state = modify_voj_separ }

	custom_effect_tooltip = SER_voj_separ_TT

	clamp_variable = {
		var = ROOT.voj_separ
		min = 0
		max = 100
	}
}

KOS_resolve_ethnic_troubles = {
	if = {
		limit = { has_idea = KOS_ethnic_troubles }
		custom_effect_tooltip = KOS_resolve_ethnic_troubles_tt
		if = {
			limit = { check_variable = { KOS_serbs < 0.25 } }
			swap_ideas = {
				remove_idea = KOS_ethnic_troubles
				add_idea = KOS_ethnic_troubles2
			}
		}
		else_if = {
			limit = { check_variable = { KOS_serbs > 0.75 } }
			swap_ideas = {
				remove_idea = KOS_ethnic_troubles
				add_idea = KOS_ethnic_troubles4
			}
		}
		else = {
			swap_ideas = {
				remove_idea = KOS_ethnic_troubles
				add_idea = KOS_ethnic_troubles3
			}
		}
	}
}
