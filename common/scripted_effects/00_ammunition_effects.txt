# Ammunition System
# Passive daily production proportional to military factory count.
# Consumption is handled natively by the low reliability (0.2) on each ammo type —
# units expend ammo naturally as they fight and operate, at ~6x the rate of infantry weapons.
#
# Production split (per factory per day):
#   Infantry ammo:  15 full / 4 reduced
#   Vehicle ammo:   12 full / 3 reduced
#   Artillery ammo: 12 full / 3 reduced
#   MLRS rockets:    6 full / 1 reduced
#   Total:          45 full / 11 reduced  (matches original aggregate)

ammunition_daily_production = {
	if = {
		limit = { num_of_military_factories > 0 }
		#TODO: Add a modifier to the strategic reserve for ammo stockpiling;
		#Nations have to have the stockpile space to store their ammo
		#If you max your storage, then additional production stops till you start consuming again
		set_temp_variable = { ammo_prod_reduced = num_of_military_factories }
		set_temp_variable = { ammo_prod_full = num_of_available_military_factories }
		subtract_from_temp_variable = { ammo_prod_reduced = ammo_prod_full }

		# Infantry ammo: 15 full / 4 reduced
		set_temp_variable = { ammo_inf_full = ammo_prod_full }
		set_temp_variable = { ammo_inf_reduced = ammo_prod_reduced }
		multiply_temp_variable = { ammo_inf_full = 15 }
		multiply_temp_variable = { ammo_inf_reduced = 4 }
		set_temp_variable = { ammo_inf_final = ammo_inf_full }
		add_to_temp_variable = { ammo_inf_final = ammo_inf_reduced }
		add_equipment_to_stockpile = {
			type = MD_infantry_ammo_1
			amount = var:ammo_inf_final
			producer = THIS
		}

		# Vehicle ammo: 12 full / 3 reduced
		set_temp_variable = { ammo_veh_full = ammo_prod_full }
		set_temp_variable = { ammo_veh_reduced = ammo_prod_reduced }
		multiply_temp_variable = { ammo_veh_full = 12 }
		multiply_temp_variable = { ammo_veh_reduced = 3 }
		set_temp_variable = { ammo_veh_final = ammo_veh_full }
		add_to_temp_variable = { ammo_veh_final = ammo_veh_reduced }
		add_equipment_to_stockpile = {
			type = MD_vehicle_ammo_1
			amount = var:ammo_veh_final
			producer = THIS
		}

		# Artillery ammo: 12 full / 3 reduced
		set_temp_variable = { ammo_art_full = ammo_prod_full }
		set_temp_variable = { ammo_art_reduced = ammo_prod_reduced }
		multiply_temp_variable = { ammo_art_full = 12 }
		multiply_temp_variable = { ammo_art_reduced = 3 }
		set_temp_variable = { ammo_art_final = ammo_art_full }
		add_to_temp_variable = { ammo_art_final = ammo_art_reduced }
		add_equipment_to_stockpile = {
			type = MD_artillery_ammo_1
			amount = var:ammo_art_final
			producer = THIS
		}

		# MLRS rockets: 6 full / 1 reduced
		set_temp_variable = { ammo_mlrs_full = ammo_prod_full }
		set_temp_variable = { ammo_mlrs_reduced = ammo_prod_reduced }
		multiply_temp_variable = { ammo_mlrs_full = 6 }
		multiply_temp_variable = { ammo_mlrs_reduced = 1 }
		set_temp_variable = { ammo_mlrs_final = ammo_mlrs_full }
		add_to_temp_variable = { ammo_mlrs_final = ammo_mlrs_reduced }
		add_equipment_to_stockpile = {
			type = MD_mlrs_ammo_1
			amount = var:ammo_mlrs_final
			producer = THIS
		}
	}
}

# Called once per country in the on_startup every_country loop.
# Seeds ~180 days of production as a starting buffer so no nation begins the game ammo-starved.
# Total per factory: 1000 + 800 + 800 + 400 = 3000 (matches original aggregate)
ammunition_startup_stockpile = {
	if = {
		limit = { num_of_military_factories > 0 }

		# Infantry ammo: 1000 per factory
		set_temp_variable = { ammo_start = num_of_military_factories }
		multiply_temp_variable = { ammo_start = 1000 }
		add_equipment_to_stockpile = {
			type = MD_infantry_ammo_1
			amount = var:ammo_start
			producer = THIS
		}

		# Vehicle ammo: 800 per factory
		set_temp_variable = { ammo_start = num_of_military_factories }
		multiply_temp_variable = { ammo_start = 800 }
		add_equipment_to_stockpile = {
			type = MD_vehicle_ammo_1
			amount = var:ammo_start
			producer = THIS
		}

		# Artillery ammo: 800 per factory
		set_temp_variable = { ammo_start = num_of_military_factories }
		multiply_temp_variable = { ammo_start = 800 }
		add_equipment_to_stockpile = {
			type = MD_artillery_ammo_1
			amount = var:ammo_start
			producer = THIS
		}

		# MLRS rockets: 400 per factory
		set_temp_variable = { ammo_start = num_of_military_factories }
		multiply_temp_variable = { ammo_start = 400 }
		add_equipment_to_stockpile = {
			type = MD_mlrs_ammo_1
			amount = var:ammo_start
			producer = THIS
		}
	}
}
