expenditure_for_mio_upgrade = {
	if = {
		limit = {
			check_variable = { free_trait_picks > 0 }
		}
		add_to_variable = { free_trait_picks = -1 }
	}
	else = {
		FROM = {
			small_expenditure = yes
		}
	}
}
