	if = { limit = { has_idea = SAU_national_water_pride_plus2 }   remove_ideas = SAU_national_water_pride_plus2 }
	if = { limit = { has_idea = SAU_national_water_pride_plus1 }   remove_ideas = SAU_national_water_pride_plus1 }
	if = { limit = { has_idea = SAU_national_water_pride_neutral } remove_ideas = SAU_national_water_pride_neutral }
	if = { limit = { has_idea = SAU_national_water_pride_minus1 }  remove_ideas = SAU_national_water_pride_minus1 }
	if = { limit = { has_idea = SAU_national_water_pride_minus2 }  remove_ideas = SAU_national_water_pride_minus2 }
	if = { limit = { has_idea = SAU_national_water_pride_minus3 }  remove_ideas = SAU_national_water_pride_minus3 }
	if = { limit = { has_idea = SAU_national_water_pride_minus4 }  remove_ideas = SAU_national_water_pride_minus4 }

	if = {
		limit = { has_stability < 0.15 }
		add_ideas = SAU_national_water_pride_minus4
		else_if = {
			limit = { has_stability < 0.25 }
			add_ideas = SAU_national_water_pride_minus3
		}
		else_if = {
			limit = { has_stability < 0.35 }
			add_ideas = SAU_national_water_pride_minus2
		}
		else_if = {
			limit = { has_stability < 0.45 }
			add_ideas = SAU_national_water_pride_minus1
		}
		else_if = {
			limit = { has_stability < 0.55 }
			add_ideas = SAU_national_water_pride_neutral
		}
		else_if = {
			limit = { has_stability < 0.65 }
			add_ideas = SAU_national_water_pride_plus1
		}
		else_if = {
			limit = { has_stability < 0.75 }
			add_ideas = SAU_national_water_pride_plus2
		}
		else_if = {
			limit = { has_stability < 0.85 }
			add_ideas = SAU_national_water_pride_plus3
		}
		else = {
			add_ideas = SAU_national_water_pride_plus4
		}
	}
}

# Strait of Hormuz tiers: open (peace), strained (Iran at war), closed (Iran at war with the USA or with us)
SAU_update_hormuz_status = {
	if = {
		limit = {
			country_exists = PER
			PER = {
				OR = {
					has_war_with = USA
					has_war_with = SAU
				}
			}
		}
		if = {
			limit = { NOT = { has_idea = SAU_strait_of_hormuz_closed } }
			if = { limit = { has_idea = SAU_strait_of_hormuz_open } remove_ideas = SAU_strait_of_hormuz_open }
			if = { limit = { has_idea = SAU_strait_of_hormuz_strained } remove_ideas = SAU_strait_of_hormuz_strained }
			add_ideas = SAU_strait_of_hormuz_closed
			country_event = { id = SAU_events.54 days = 1 }
			news_event = { id = SAU_events.66 days = 1 }
		}
	}
	else_if = {
		limit = {
			country_exists = PER
			PER = { has_war = yes }
		}
		if = {
			limit = { has_idea = SAU_strait_of_hormuz_closed }
			remove_ideas = SAU_strait_of_hormuz_closed
			country_event = { id = SAU_events.55 days = 1 }
		}
		if = { limit = { has_idea = SAU_strait_of_hormuz_open } remove_ideas = SAU_strait_of_hormuz_open }
		if = {
			limit = { NOT = { has_idea = SAU_strait_of_hormuz_strained } }
			add_ideas = SAU_strait_of_hormuz_strained
		}
	}
	else = {
		if = {
			limit = { has_idea = SAU_strait_of_hormuz_closed }
			remove_ideas = SAU_strait_of_hormuz_closed
			country_event = { id = SAU_events.55 days = 1 }
		}
		if = { limit = { has_idea = SAU_strait_of_hormuz_strained } remove_ideas = SAU_strait_of_hormuz_strained }
		if = {
			limit = { NOT = { has_idea = SAU_strait_of_hormuz_open } }
			add_ideas = SAU_strait_of_hormuz_open
		}
	}
}

# Re-evaluate the Vision 2030 national mood each month from projects delivered
SAU_update_vision2030 = {
	set_temp_variable = { v2030_eff = SAU_giga_projects_built }
	subtract_from_temp_variable = { v2030_eff = SAU_giga_projects_failed }
	if = {
		limit = { check_variable = { v2030_eff > 11 } }
		if = {
			limit = { NOT = { has_idea = SAU_vision_2030_delivered } }
			SAU_clear_vision_ideas = yes
			add_ideas = SAU_vision_2030_delivered
			news_event = { id = SAU_events.82 days = 1 }
		}
	}
	else_if = {
		limit = {
			OR = {
				AND = { date > 2030.1.1 check_variable = { v2030_eff < 6 } }
				check_variable = { SAU_giga_projects_failed > 7 }
			}
		}
		if = {
			limit = { NOT = { has_idea = SAU_vision_2030_white_elephant } }
			SAU_clear_vision_ideas = yes
			add_ideas = SAU_vision_2030_white_elephant
			country_event = { id = SAU_events.83 days = 1 }
		}
	}
	else_if = {
		limit = { check_variable = { v2030_eff > 5 } }
		if = {
			limit = { NOT = { has_idea = SAU_vision_2030_progress } }
			SAU_clear_vision_ideas = yes
			add_ideas = SAU_vision_2030_progress
		}
	}
	else = {
		if = {
			limit = { NOT = { has_idea = SAU_vision_2030_expectations } }
			SAU_clear_vision_ideas = yes
			add_ideas = SAU_vision_2030_expectations
		}
	}
}
