
on_actions = {
	# Taiwan election result — affects CHI_TAI_trust
	on_ruling_party_change = {
		effect = {
			# Pan-Green (DPP, liberalism index 2) wins → trust penalty
			if = {
				limit = {
					original_tag = TAI
					is_in_array = { ruling_party = 2 }
					CHI = {
						exists = yes
						check_variable = { CHI_TAI_trust > 0 }
					}
				}
				CHI = {
					add_to_variable = { CHI_TAI_trust = -22 }
					if = {
						limit = { check_variable = { CHI_TAI_trust < 0 } }
						set_variable = { CHI_TAI_trust = 0 }
					}
					country_event = { id = china_sar.26 days = 7 }
				}
			}
			# Pan-Blue (KMT conservatism index 1 or Neutral_conservatism index 14) wins → trust bonus
			else_if = {
				limit = {
					original_tag = TAI
					OR = {
						is_in_array = { ruling_party = 1 }
						is_in_array = { ruling_party = 14 }
					}
					CHI = { exists = yes }
				}
				CHI = { add_to_variable = { CHI_TAI_trust = 5 } }
			}
		}
	}

	on_army_leader_won_combat = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = allow_easter_eggs
						option = yes
					}
				}
				if = {
					limit = {
						FROM = {
							original_tag = CHI
							OR = {
								has_country_flag = {
									flag = CHI_cnc_general_quote_used
									days > 30
								}
								NOT = { has_country_flag = CHI_cnc_general_quote_used }
							}
						}
					}
					if = {
						limit = { ROOT = { has_trait = CHI_special_weapons_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_leang_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_infantry_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_fai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_tank_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_kwai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_nuke_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_tao_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
				}
			}
		}
	}
	on_unit_leader_level_up = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = allow_easter_eggs
						option = yes
					}
				}
				if = {
					limit = { FROM = { original_tag = CHI } }
					if = {
						limit = { ROOT = { has_trait = CHI_special_weapons_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_leang_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_infantry_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_fai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_tank_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_kwai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { ROOT = { has_trait = CHI_nuke_general } }
						FROM = {
							scoped_sound_effect = "CHI_general_tao_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
				}
			}
		}
	}
}
