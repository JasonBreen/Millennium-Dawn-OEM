
on_actions = {
	# Taiwan election result — affects CHI_TAI_trust
	on_ruling_party_change = {
		effect = {
			# Pan-Green (DPP, liberalism index 2) wins → trust penalty
			if = {
				limit = {
					original_tag = TAI
					is_in_array = { ruling_party = 2 }
					CHI = {
						exists = yes
						check_variable = { CHI_TAI_trust > 0 }
					}
				}
				CHI = {
					add_to_variable = { CHI_TAI_trust = -22 }
					if = {
						limit = { check_variable = { CHI_TAI_trust < 0 } }
						set_variable = { CHI_TAI_trust = 0 }
					}
					country_event = { id = china_sar.004 days = 7 }
				}
			}
			# Pan-Blue (KMT conservatism index 1 or Neutral_conservatism index 14) wins → trust bonus
			else_if = {
				limit = {
					original_tag = TAI
					OR = {
						is_in_array = { ruling_party = 1 }
						is_in_array = { ruling_party = 14 }
					}
					CHI = { exists = yes }
				}
				CHI = { add_to_variable = { CHI_TAI_trust = 5 } }
			}

			### US president change — affects CHI_taiwan_us_support
			if = {
				limit = {
					original_tag = USA
					CHI = {
						exists = yes
						has_country_flag = CHI_TAI_diplomatic_route_active
					}
				}
				# Clear previous president flags
				CHI = { clr_country_flag = CHI_trump_is_president }
				CHI = { clr_country_flag = CHI_harris_us_support_floor }

				# Pro-Taiwan presidents — increase us_support
				if = {
					limit = { USA = { has_country_leader = { name = "John McCain" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 10 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Joe Biden" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 8 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Hillary Clinton" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 6 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Kamala Harris" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 6 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						set_country_flag = CHI_harris_us_support_floor
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "George W. Bush" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 5 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Barack Obama" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 5 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Mitt Romney" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 4 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Al Gore" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = 3 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				# Anti-Taiwan / manipuleerbare presidenten — decrease us_support
				else_if = {
					limit = { USA = { has_country_leader = { name = "John Kerry" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = -2 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						country_event = { id = china_sar.026 days = 3 }
					}
				}
				else_if = {
					limit = { USA = { has_country_leader = { name = "Donald Trump" } } }
					CHI = {
						add_to_variable = { CHI_taiwan_us_support = -10 tooltip = CHI_taiwan_us_support_tt }
						clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
						set_country_flag = CHI_trump_is_president
						country_event = { id = china_sar.026 days = 3 }
					}
				}
			}
		}
	}

	### Taiwan Strait — monthly checks
	on_monthly = {
		effect = {
			if = {
				limit = {
					original_tag = CHI
					has_country_flag = CHI_TAI_diplomatic_route_active
					country_exists = TAI
					NOT = { has_war_with = TAI }
				}

				# Tension decay: -1 per month if no recent provocation
				if = {
					limit = {
						NOT = { has_country_flag = CHI_recent_provocation }
						check_variable = { CHI_taiwan_strait_tension > 0 }
					}
					add_to_variable = { CHI_taiwan_strait_tension = -1 }
					clamp_variable = { var = CHI_taiwan_strait_tension min = 0 max = 100 }
				}

				# US support passive decay: -1 every other month if economic leverage active
				if = {
					limit = {
						has_completed_focus = CHI_TAI_economic_leverage
						has_country_flag = { flag = CHI_us_support_decay_tick days > 60 }
						check_variable = { CHI_taiwan_us_support > 0 }
					}
					add_to_variable = { CHI_taiwan_us_support = -1 }
					clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
					clr_country_flag = CHI_us_support_decay_tick
					set_country_flag = CHI_us_support_decay_tick
				}
				if = {
					limit = {
						has_completed_focus = CHI_TAI_economic_leverage
						NOT = { has_country_flag = CHI_us_support_decay_tick }
					}
					set_country_flag = CHI_us_support_decay_tick
				}

				# Harris floor: us_support cannot drop below 40 while Harris is president
				if = {
					limit = {
						has_country_flag = CHI_harris_us_support_floor
						check_variable = { CHI_taiwan_us_support < 40 }
					}
					set_variable = { CHI_taiwan_us_support = 40 }
				}

				# Crisis trigger: tension reaches 100
				if = {
					limit = {
						check_variable = { CHI_taiwan_strait_tension > 99 }
						NOT = { has_country_flag = CHI_taiwan_crisis_triggered }
					}
					set_country_flag = CHI_taiwan_crisis_triggered
					country_event = { id = china_sar.019 days = 1 }
				}

				# Diplomatic isolation milestone: recognition below 5
				if = {
					limit = {
						check_variable = { CHI_taiwan_recognition_count < 5 }
						NOT = { has_country_flag = CHI_taiwan_diplomatic_alone }
					}
					set_country_flag = CHI_taiwan_diplomatic_alone
					country_event = { id = china_sar.011 days = 1 }
				}

				# USA at war with multiple nations — us_support penalty
				if = {
					limit = {
						NOT = { has_country_flag = CHI_usa_multiwar_penalty_applied }
						USA = {
							has_war = yes
							num_of_factories > 0
							count_triggers = {
								amount = 2
								has_war_with = JAP
								has_war_with = SOV
								has_war_with = CHI
								has_war_with = NKO
								has_war_with = PER
								has_war_with = IRQ
								has_war_with = AFG
								has_war_with = SYR
							}
						}
					}
					set_country_flag = CHI_usa_multiwar_penalty_applied
					add_to_variable = { CHI_taiwan_us_support = -8 tooltip = CHI_taiwan_us_support_tt }
					clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
				}

				# Lobby network periodic events
				if = {
					limit = {
						has_completed_focus = CHI_TAI_lobby_network
						has_country_flag = { flag = CHI_lobby_event_cooldown days > 180 }
					}
					clr_country_flag = CHI_lobby_event_cooldown
					set_country_flag = CHI_lobby_event_cooldown
					random_list = {
						50 = { country_event = { id = china_sar.012 days = 3 } }
						50 = { country_event = { id = china_sar.013 days = 3 } }
					}
				}
				if = {
					limit = {
						has_completed_focus = CHI_TAI_lobby_network
						NOT = { has_country_flag = CHI_lobby_event_cooldown }
					}
					set_country_flag = CHI_lobby_event_cooldown
				}

				# Blockade monthly effects
				if = {
					limit = { has_country_flag = CHI_TAI_full_blockade_active }
					add_to_variable = { CHI_taiwan_strait_tension = 2 tooltip = CHI_taiwan_strait_tension_tt }
					clamp_variable = { var = CHI_taiwan_strait_tension min = 0 max = 100 }
					add_to_variable = { CHI_taiwan_us_support = -3 tooltip = CHI_taiwan_us_support_tt }
					clamp_variable = { var = CHI_taiwan_us_support min = 0 max = 100 }
				}
			}
		}
	}

	on_annex = {
		effect = {
			# When CHI annexes TAI, clean up the Taiwan aggression system
			if = {
				limit = {
					ROOT = { original_tag = CHI }
					FROM = { original_tag = TAI }
					ROOT = { has_completed_focus = CHI_TAI_combat_the_traitors }
				}
				ROOT = {
					remove_dynamic_modifier = { modifier = CHI_TAI_shared_modifier }
					clr_country_flag = CHI_TAI_diplomatic_route_active
					clr_country_flag = CHI_taiwan_crisis_triggered
					clr_country_flag = CHI_taiwan_diplomatic_alone
					clr_country_flag = CHI_TAI_full_blockade_active
					clr_country_flag = CHI_recent_provocation
				}
			}

			# NPM annexed NEP under Chinese sponsorship - settlement event
			if = {
				limit = {
					ROOT = { original_tag = NPM }
					FROM = { original_tag = NEP }
					CHI = {
						exists = yes
						has_country_flag = CHI_backing_npm_insurgency
					}
				}
				ROOT = {
					remove_ideas = CHI_npm_red_banner_insurgency
					country_event = { id = sino_indian.53 days = 3 }
				}
				CHI = { clr_country_flag = CHI_backing_npm_insurgency }
			}
		}
	}

	### SCO Yearly Cohesion Cap Reset ###

					original_tag = CHI
					has_completed_focus = CHI_Form_the_SCO
					NOT = { has_country_flag = sco_yearly_cohesion_reset }

	### SCO Variable Threshold Monitoring ###
	on_monthly = {
		effect = {
			if = {
				limit = {
					original_tag = CHI
					has_completed_focus = CHI_Form_the_SCO
				}

				# Low cohesion warning
				if = {
					limit = {
						check_variable = { CHI_sco_cohesion < 2 }
						NOT = { has_country_flag = sco_low_cohesion_fired }
					}
					set_country_flag = { flag = sco_low_cohesion_fired days = 365 value = 1 }
					country_event = { id = CHI_sco.20 days = 1 }
				}

				# High cohesion milestone
				if = {
					limit = {
						check_variable = { CHI_sco_cohesion > 8 }
						NOT = { has_country_flag = sco_high_cohesion_fired }
					}
					set_country_flag = { flag = sco_high_cohesion_fired days = 365 value = 1 }
					country_event = { id = CHI_sco.21 days = 1 }
				}

				# China dominant — balance > 5
				if = {
					limit = {
						check_variable = { CHI_sco_balance > 5 }
						NOT = { has_country_flag = sco_china_dominant_fired }
					}
					set_country_flag = { flag = sco_china_dominant_fired days = 365 value = 1 }
					country_event = { id = CHI_sco.22 days = 1 }
				}

				# Russia dominant — balance < -5
				if = {
					limit = {
						check_variable = { CHI_sco_balance < -5 }
						NOT = { has_country_flag = sco_russia_dominant_fired }
					}
					set_country_flag = { flag = sco_russia_dominant_fired days = 365 value = 1 }
					country_event = { id = CHI_sco.23 days = 1 }
				}
			}
		}
	}

	### Japan branch rework — Senkaku Pressure tick & threshold incidents ###
	on_monthly = {
		effect = {
			if = {
				limit = {
					original_tag = CHI
					has_country_flag = CHI_paf_militia_toggle_active
					country_exists = JAP
					NOT = { has_war_with = JAP }
				}
				add_to_variable = { senkaku_pressure = 2 tooltip = senkaku_pressure_tt }
				clamp_variable = { var = senkaku_pressure min = 0 max = 100 }

				# Threshold 30 — first incident
				if = {
					limit = {
						check_variable = { senkaku_pressure > 29 }
						NOT = { has_country_flag = senkaku_incident_30_fired }
					}
					set_country_flag = senkaku_incident_30_fired
					JAP = { country_event = { id = CHI_foreign.212 days = 3 } }
				}
				# Threshold 60 — second incident
				if = {
					limit = {
						check_variable = { senkaku_pressure > 59 }
						NOT = { has_country_flag = senkaku_incident_60_fired }
					}
					set_country_flag = senkaku_incident_60_fired
					JAP = { country_event = { id = CHI_foreign.212 days = 3 } }
				}
				# Threshold 90 — third and final pre-crisis incident
				if = {
					limit = {
						check_variable = { senkaku_pressure > 89 }
						NOT = { has_country_flag = senkaku_incident_90_fired }
					}
					set_country_flag = senkaku_incident_90_fired
					JAP = { country_event = { id = CHI_foreign.212 days = 3 } }
				}
			}
		}
	}
