on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { original_tag = SAU }
				country_event = { id = SAU_startup.1 }
				if = {
					limit = { NOT = { has_country_flag = SAU_oil_dependence_set } }
					set_variable = { oil_dependence = 90 }
					set_country_flag = SAU_oil_dependence_set
				}
			}
		}
	}

	on_monthly = {
		effect = {
			if = {
			limit = { tag = SAU }
			SAU = {
				if = {
					limit = {
						OR = {
							has_idea = SAU_national_water_pride_plus4
							has_idea = SAU_national_water_pride_plus3
							has_idea = SAU_national_water_pride_plus2
							has_idea = SAU_national_water_pride_plus1
							has_idea = SAU_national_water_pride_neutral
							has_idea = SAU_national_water_pride_minus1
							has_idea = SAU_national_water_pride_minus2
							has_idea = SAU_national_water_pride_minus3
							has_idea = SAU_national_water_pride_minus4
						}
					}
					if = { limit = { has_idea = SAU_national_water_pride_plus4 }   remove_ideas = SAU_national_water_pride_plus4 }
					if = { limit = { has_idea = SAU_national_water_pride_plus3 }   remove_ideas = SAU_national_water_pride_plus3 }
					if = { limit = { has_idea = SAU_national_water_pride_plus2 }   remove_ideas = SAU_national_water_pride_plus2 }
					if = { limit = { has_idea = SAU_national_water_pride_plus1 }   remove_ideas = SAU_national_water_pride_plus1 }
					if = { limit = { has_idea = SAU_national_water_pride_neutral } remove_ideas = SAU_national_water_pride_neutral }
					if = { limit = { has_idea = SAU_national_water_pride_minus1 }  remove_ideas = SAU_national_water_pride_minus1 }
					if = { limit = { has_idea = SAU_national_water_pride_minus2 }  remove_ideas = SAU_national_water_pride_minus2 }
					if = { limit = { has_idea = SAU_national_water_pride_minus3 }  remove_ideas = SAU_national_water_pride_minus3 }
					if = { limit = { has_idea = SAU_national_water_pride_minus4 }  remove_ideas = SAU_national_water_pride_minus4 }
					if = { limit = { has_stability < 0.15 } add_ideas = SAU_national_water_pride_minus4 }
					if = { limit = { NOT = { has_stability < 0.15 } has_stability < 0.25 } add_ideas = SAU_national_water_pride_minus3 }
					if = { limit = { NOT = { has_stability < 0.25 } has_stability < 0.35 } add_ideas = SAU_national_water_pride_minus2 }
					if = { limit = { NOT = { has_stability < 0.35 } has_stability < 0.45 } add_ideas = SAU_national_water_pride_minus1 }
					if = { limit = { NOT = { has_stability < 0.45 } has_stability < 0.55 } add_ideas = SAU_national_water_pride_neutral }
					if = { limit = { NOT = { has_stability < 0.55 } has_stability < 0.65 } add_ideas = SAU_national_water_pride_plus1 }
					if = { limit = { NOT = { has_stability < 0.65 } has_stability < 0.75 } add_ideas = SAU_national_water_pride_plus2 }
					if = { limit = { NOT = { has_stability < 0.75 } has_stability < 0.85 } add_ideas = SAU_national_water_pride_plus3 }
					if = { limit = { NOT = { has_stability < 0.85 } } add_ideas = SAU_national_water_pride_plus4 }
				}


				if = {
					limit = {
						has_idea = wahabi_ulema
						NOT = {
							OR = {
								has_idea = SAU_royal_ulema_alliance_high
								has_idea = SAU_royal_ulema_alliance_content
								has_idea = SAU_royal_ulema_alliance_neutral
								has_idea = SAU_royal_ulema_alliance_strained
							}
						}
					}
					add_ideas = SAU_royal_ulema_alliance_neutral
				}
				if = {
					limit = {
						OR = {
							has_idea = SAU_royal_ulema_alliance_high
							has_idea = SAU_royal_ulema_alliance_content
							has_idea = SAU_royal_ulema_alliance_neutral
							has_idea = SAU_royal_ulema_alliance_strained
						}
					}
					if = { limit = { has_idea = SAU_royal_ulema_alliance_high }     remove_ideas = SAU_royal_ulema_alliance_high }
					if = { limit = { has_idea = SAU_royal_ulema_alliance_content }  remove_ideas = SAU_royal_ulema_alliance_content }
					if = { limit = { has_idea = SAU_royal_ulema_alliance_neutral }  remove_ideas = SAU_royal_ulema_alliance_neutral }
					if = { limit = { has_idea = SAU_royal_ulema_alliance_strained } remove_ideas = SAU_royal_ulema_alliance_strained }
					if = {
						limit = { check_variable = { wahabi_ulema_opinion > 75 } }
						add_ideas = SAU_royal_ulema_alliance_high
					}
					if = {
						limit = {
							check_variable = { wahabi_ulema_opinion > 50 }
							check_variable = { wahabi_ulema_opinion < 76 }
						}
						add_ideas = SAU_royal_ulema_alliance_content
					}
					if = {
						limit = {
							check_variable = { wahabi_ulema_opinion > 24 }
							check_variable = { wahabi_ulema_opinion < 51 }
						}
						add_ideas = SAU_royal_ulema_alliance_neutral
					}
					if = {
						limit = { check_variable = { wahabi_ulema_opinion < 25 } }
						add_ideas = SAU_royal_ulema_alliance_strained
					}
				}

				clamp_variable = { var = oil_dependence min = 0 max = 100 }

				if = {
					limit = { check_variable = { oil_dependence < 100 } }
					country_event = { id = SAU_economy.4 }
				}

				if = {
					limit = {
						OR = {
							has_idea = SAU_oil_dependence_t5
							has_idea = SAU_oil_dependence_t4
							has_idea = SAU_oil_dependence_t3
							has_idea = SAU_oil_dependence_t2
							has_idea = SAU_oil_dependence_t1
						}
					}
					if = { limit = { has_idea = SAU_oil_dependence_t5 } remove_ideas = SAU_oil_dependence_t5 }
					if = { limit = { has_idea = SAU_oil_dependence_t4 } remove_ideas = SAU_oil_dependence_t4 }
					if = { limit = { has_idea = SAU_oil_dependence_t3 } remove_ideas = SAU_oil_dependence_t3 }
					if = { limit = { has_idea = SAU_oil_dependence_t2 } remove_ideas = SAU_oil_dependence_t2 }
					if = { limit = { has_idea = SAU_oil_dependence_t1 } remove_ideas = SAU_oil_dependence_t1 }
					if = { limit = { check_variable = { oil_dependence > 80 } } add_ideas = SAU_oil_dependence_t5 }
					if = { limit = { check_variable = { oil_dependence > 60 } check_variable = { oil_dependence < 80.001 } } add_ideas = SAU_oil_dependence_t4 }
					if = { limit = { check_variable = { oil_dependence > 40 } check_variable = { oil_dependence < 60.001 } } add_ideas = SAU_oil_dependence_t3 }
					if = { limit = { check_variable = { oil_dependence > 20 } check_variable = { oil_dependence < 40.001 } } add_ideas = SAU_oil_dependence_t2 }
					if = { limit = { check_variable = { oil_dependence < 20.001 } } add_ideas = SAU_oil_dependence_t1 }
				}
			}
			}
		
			if = {
				limit = {
					tag = SAU
					has_completed_focus = SAU_Freedom_or_deat
					NOT = { has_completed_focus = SAU_Total_Victory_in_Syria }
					NOT = { has_completed_focus = SAU_Total_lost_in_Syria }
				}
				if = {
					limit = {
						OR = {
							NOT = { country_exists = SYR }
							SYR = { has_capitulated = yes }
						}
					}
					complete_national_focus = SAU_Total_Victory_in_Syria
				}
				else_if = {
					limit = {
						country_exists = SYR
						NOT = { SYR = { has_capitulated = yes } }
						OR = {
							NOT = { country_exists = FSA }
							FSA = { has_capitulated = yes }
						}
					}
					complete_national_focus = SAU_Total_lost_in_Syria
				}
			}
		
			if = {
				limit = {
					tag = SAU
					has_country_flag = SAU_yemen_pipeline_open
				}
				if = {
					limit = { SAU_houthis_won_south = yes }
					country_event = { id = SAU_events.28 hours = 1 }
				}
				else_if = {
					limit = { SAU_yemen_at_war = yes }
					random = {
						chance = 15
						country_event = { id = SAU_events.28 hours = 1 }
					}
				}
			}
		
			
			if = {
				limit = {
					tag = SAU
					country_exists = HOU
					country_exists = YEM
					HOU = { has_country_flag = yemen_partition_north }
					HOU = { has_idea = faction_resistance_axis_member }
				}
				HOU = { remove_ideas = faction_resistance_axis_member }
			}
		}
	}

	on_daily = {
		effect = {
			if = {
				limit = {
					tag = SAU
					has_war = no
					no_jihadist_government = yes
					NOT = { has_country_flag = SAU_bop_event_cooldown }
				}
				random_list = {
					1 = {
						set_country_flag = { flag = SAU_bop_event_cooldown value = 1 days = 30 }
						random_list = {
							1 = { country_event = { id = SAU_bop_events.1 } }
							1 = { country_event = { id = SAU_bop_events.2 } }
							1 = { country_event = { id = SAU_bop_events.3 } }
							1 = { country_event = { id = SAU_bop_events.4 } }
						}
					}
					99 = { }
				}
			}
		}
	}

	on_weekly = {
		effect = {
			if = {
				limit = { original_tag = SAU }
				SAU_update_hormuz_status = yes
			}
		}
	}
}
