on_actions = {
	on_weekly_GRE = {
		effect = {
			# Process creditor interest payments
			if = {
				limit = { check_variable = { GRE_creditors^num > 0 } }
				GRE_process_creditor_payments = yes
			}

			# Recalculate unified debt ratio (factors generic debt + creditor debt at 0.5x)
			GRE_calculate_unified_debt = yes
		}
	}
}
