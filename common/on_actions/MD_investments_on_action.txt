on_actions = {
    on_weekly = {
        effect = {
            random_state = {
                limit = {
                    NOT = {
                        state = ROOT
                    }
                }

                set_temp_variable = { state = THIS }
                set_temp_variable_to_random = {
                    var = building_index
                    min = 1
                    max = 5
                    integer = yes
                }
                set_temp_variable_to_random = {
                    var = amount
                    min = 1
                    max = 3
                    integer = yes
                }
                ROOT = {
                    create_investment_project = yes
                }
            }
        }
    }
	on_daily = {
        effect = {
            for_each_loop = {
                array = investments_projects_building_time
                index = i
                value = v

                subtract_from_variable = { investments_projects_building_time^i = 1 }

                if = {
                    limit = {
                        check_variable = { investments_projects_building_time^i < 1 }
                    }

                    var:investments_projects_state^i = {
                        add_building_construction = {
                            type = arms_factory
                            level = 1
                            instant_build = yes
                        }

                        owner = {
                            set_temp_variable = { country = ROOT }
                            set_temp_variable = { amount = 10 }
                            add_investment = yes
                        }
                    }

                    set_temp_variable = { building_index = investments_projects_building^i }
                    set_temp_variable = { amount = 1 }
                    set_temp_variable = { state = investments_projects_state^i }
                    remove_investment_project = yes
                }
            }
        }
    }
}
