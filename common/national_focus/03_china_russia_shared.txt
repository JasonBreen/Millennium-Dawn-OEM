			exists = yes

				target = SOV
				modifier = mutual_trade_agreement
			}

				target = CHI
				modifier = mutual_trade_agreement
			}
		}
		else = {
			add_opinion_modifier = {
				target = CHI
				modifier = mutual_trade_agreement
			}
			reverse_add_opinion_modifier = {
				target = SOV
				modifier = mutual_trade_agreement
			}
		}
	}

	ai_will_do = {
		base = 5
		modifier = {
			factor = 0
			original_tag = SOV
		}
