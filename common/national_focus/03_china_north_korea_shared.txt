	id = CHI_NKO_bilateral_cooperation
	icon = consumer_goods
	text_icon = JOINT_NKO_CHI_focus_style
	x = 60
	y = 20

	cost = 5

	offset = {
		x = 10
		y = -20
		trigger = { original_tag = NKO }
	}

	search_filters = { FOCUS_FILTER_POLITICAL }

	available = {
		CHI = {
			exists = yes
			NOT = { has_war_with = NKO }
			NOT = { has_completed_focus = CHI_End_the_Kim_Dynasty }
		}
		NKO = {
			exists = yes
			OR = {
				is_subject = no
				is_subject_of = CHI
			}
		}
	}

	joint_trigger = {
		OR = {
			tag = CHI
			tag = NKO
		}
	}

	completion_reward = {
		log = "[GetDateText]: [Root.GetName]: Focus CHI_NKO_bilateral_cooperation"
		add_political_power = 25
		if = {
			limit = { original_tag = CHI }
			add_opinion_modifier = {
				target = NKO
				modifier = mutual_trade_agreement
			}
			reverse_add_opinion_modifier = {
				target = CHI
				modifier = mutual_trade_agreement
			}
		}
		else = {
			add_opinion_modifier = {
				target = CHI
				modifier = mutual_trade_agreement
			}
			reverse_add_opinion_modifier = {
				target = NKO
				modifier = mutual_trade_agreement
			}
		}
	}

	ai_will_do = {
		base = 5
		modifier = {
			factor = 0
			original_tag = NKO
		}
	}
}
