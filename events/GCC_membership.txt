add_namespace = gcc_invite

# Invitation to join the Gulf Cooperation Council
# Fired to the candidate nation. FROM = the inviting Dominant Member.
country_event = {
	id = gcc_invite.1
	title = gcc_invite.1.t
	desc = gcc_invite.1.d
	picture = GFX_SAU_generic

	is_triggered_only = yes

	option = {
		name = gcc_invite.1.a		# Accept
		ai_chance = {
			base = 5
			modifier = { add = 60 has_opinion = { target = FROM value > 40 } }
			modifier = { add = 25 has_opinion = { target = FROM value > 10 } }
			modifier = { factor = 0 has_opinion = { target = FROM value < 0 } }
			modifier = { factor = 0 has_war_with = FROM }
		}
		add_ideas = idea_gcc_member_state
		set_temp_variable = { temp_modify_gcc_unity = 5 }
		modify_gcc_unity = yes
		every_country = {
			limit = {
				has_idea = idea_gcc_member_state
				NOT = { tag = ROOT }
			}
			add_opinion_modifier = { target = ROOT modifier = GCC_new_member }
			ROOT = { add_opinion_modifier = { target = PREV modifier = GCC_new_member } }
		}
		news_event = { id = gcc_invite.2 days = 1 }
	}
	option = {
		name = gcc_invite.1.b		# Refuse
		ai_chance = {
			base = 5
			modifier = { factor = 0.2 has_opinion = { target = FROM value > 40 } }
		}
		FROM = { country_event = { id = gcc_invite.3 days = 1 } }
	}
}

# News: a new member joins the GCC (broadcast)
news_event = {
	id = gcc_invite.2
	title = gcc_invite.2.t
	desc = gcc_invite.2.d
	picture = GFX_news_gcc

	major = yes
	is_triggered_only = yes

	option = {
		name = gcc_invite.2.a
	}
}
