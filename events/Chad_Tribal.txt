# Zaghawa Protests (Level 1)
country_event = {
	id = chad_tribal.1
	title = chad_tribal.1.t
	desc = chad_tribal.1.desc
	picture = GFX_african_demonstration

	trigger = {
		original_tag = CHA
		CHA_zaghawa_protest_level = yes
		NOT = { has_country_flag = CHA_zaghawa_protest_warning }
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		set_country_flag = CHA_zaghawa_protest_warning
	}

	option = {
		name = chad_tribal.1.a
		log = "[GetDateText]: [Root.GetName]: chad_tribal.1.a executed"
		add_political_power = -50
		add_to_variable = { CHA_zaghawa_loyalty = 10 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		CHA_update_tribal_resource_modifiers = yes
		ai_chance = { base = 70 }
	}

	option = {
		name = chad_tribal.1.b
		log = "[GetDateText]: [Root.GetName]: chad_tribal.1.b executed"
		add_stability = -0.03
		add_to_variable = { CHA_zaghawa_loyalty = -5 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		CHA_update_tribal_resource_modifiers = yes
		add_timed_idea = { idea = CHA_tribal_protests days = 180 }
		ai_chance = { base = 30 }
	}
}
