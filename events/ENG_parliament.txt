# ============================================================
# ELECTION RESULTS
# Displays all seats, then chains to appropriate formation event
# ============================================================
country_event = {
	id = ENG_parliament.2
	title = ENG_parliament.2.t
	desc = ENG_parliament.2.d
	picture = GFX_report_event_generic_parliament
	is_triggered_only = yes

	# Majority government - non extremist
	option = {
		name = ENG_parliament.2.a
		log = "[GetDateText]: [This.GetName]: ENG_parliament.2.a executed"
		trigger = {
			check_variable = { ENG_hoc_majority = 1 }
			check_variable = { ENG_hoc_extremist_winner = 0 }
		}
		# Strong majority (>340)
		if = {
			limit = { check_variable = { ENG_hoc_ruling_seats > 340 } }
			add_ideas = ENG_strong_majority_spirit
		}
		else = {
			add_ideas = ENG_slim_majority_spirit
		}
		ai_chance = { base = 1 }
	}

	# Hung parliament - no majority
	option = {
		name = ENG_parliament.2.b
		log = "[GetDateText]: [This.GetName]: ENG_parliament.2.b executed"
		trigger = {
			check_variable = { ENG_hoc_majority = 0 }
			check_variable = { ENG_hoc_extremist_winner = 0 }
		}
		country_event = { id = ENG_parliament.20 days = 5 }
		ai_chance = { base = 1 }
	}

	# Extremist wins - monarchy intervention
	option = {
		name = ENG_parliament.2.c
		log = "[GetDateText]: [This.GetName]: ENG_parliament.2.c executed"
		trigger = {
			check_variable = { ENG_hoc_extremist_winner = 1 }
		}
		country_event = { id = ENG_parliament.3 days = 3 }
		ai_chance = { base = 1 }
	}
}

# MONARCHY BLOCKS EXTREMIST PM
# ============================================================
country_event = {
	id = ENG_parliament.3
	title = ENG_parliament.3.t
	desc = ENG_parliament.3.d
	picture = GFX_report_event_generic_parliament
	is_triggered_only = yes

	# Accept monarch's choice - next largest non-extremist party
	option = {
		name = ENG_parliament.3.a
		log = "[GetDateText]: [This.GetName]: ENG_parliament.3.a executed"
		custom_effect_tooltip = ENG_hoc_accept_monarch_TT
		# The monarch picks the next largest non-extremist party
		# This effectively means the 2nd place party governs
		if = {
			limit = { check_variable = { ENG_hoc_majority = 1 } }
			add_ideas = ENG_slim_majority_spirit
		}
		else = {
			country_event = { id = ENG_parliament.20 days = 3 }
		}
		ai_chance = { base = 9 }
	}

	# Defy the crown - play as the extremist party
	option = {
		name = ENG_parliament.3.b
		log = "[GetDateText]: [This.GetName]: ENG_parliament.3.b executed"
		custom_effect_tooltip = ENG_hoc_defy_monarch_TT
		set_country_flag = ENG_hoc_extremist_pm
		add_ideas = ENG_constitutional_crisis
		ai_chance = { base = 1 }
	}
}

# HUNG PARLIAMENT - Choose coalition or minority
# ============================================================
country_event = {
	id = ENG_parliament.20
	title = ENG_parliament.20.t
	desc = ENG_parliament.20.d
	picture = GFX_report_event_generic_parliament
	is_triggered_only = yes

	# Seek coalition with Lib Dems
	option = {
		name = ENG_parliament.20.a
		log = "[GetDateText]: [This.GetName]: ENG_parliament.20.a executed"
		trigger = {
			check_variable = { ENG_hoc_seats^2 > 5 }
		}
		country_event = { id = ENG_parliament.30 days = 5 }
		ai_chance = { base = 5 }
	}

	# Seek coalition with SNP
	option = {
		name = ENG_parliament.20.b
		log = "[GetDateText]: [This.GetName]: ENG_parliament.20.b executed"
		trigger = {
			check_variable = { ENG_hoc_snp_seats > 15 }
			NOT = { country_exists = SCO }
		}
		country_event = { id = ENG_parliament.31 days = 5 }
		ai_chance = {
			base = 3
			# AI prefers this if Labour is ruling party
			modifier = {
				factor = 2
				check_variable = { ENG_hoc_largest_party = 3 }
			}
		}
	}

	# Seek coalition with DUP
	option = {
		name = ENG_parliament.20.c
		log = "[GetDateText]: [This.GetName]: ENG_parliament.20.c executed"
		trigger = {
			check_variable = { ENG_hoc_dup_seats > 3 }
			NOT = { country_exists = NIR }
		}
		country_event = { id = ENG_parliament.32 days = 5 }
		ai_chance = {
			base = 2
			# AI prefers this if Conservatives are ruling party
			modifier = {
				factor = 3
				check_variable = { ENG_hoc_largest_party = 1 }
			}
		}
	}

	# Minority government
	option = {
		name = ENG_parliament.20.d
		log = "[GetDateText]: [This.GetName]: ENG_parliament.20.d executed"
		set_country_flag = ENG_hoc_hung_parliament
		add_ideas = ENG_minority_government_spirit
		ai_chance = { base = 1 }
	}
}
