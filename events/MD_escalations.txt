#  Escalation Events  #
##########################

# Purpose of escalations: To give content to regions and countries that otherwise are neglected, and to make the world feel more alive.
# Escalations will also provide a more "diplomatic" feel for players interacting with the event chains
# These wars are meant to be small, objective oriented wars with the possibility of obviously letting the player win in totality. But the purpose is to settle disputes, etc.
# Escalations can either be settled peacefully, or fought out
# Escalations will involve countries from smaller to larger as time advances
# Depending on internal factions, they can demand certain outcomes or pressure you to take certain decisions over others
# People can disable escalations via game rules

add_namespace = escalations
add_namespace = escalations_global

# Oil War of the Atlantic (Guyana V. Suriname) - 2000
# Guyana, backed by private security and tacit UK support, sends a naval escort to retake the rig. Suriname appeals to Brazil for "regional stability" protection

country_event = { # Suriname: Guyanese Oil Drilling Boat in Our Waters!
	id = escalations.1
	title = escalations.1.t
	desc = escalations.1.d

	is_triggered_only = yes

	option = { # Drive it out!
		name = escalations.1.a
		log = "[GetDateText]: [This.GetName]: bashkiriya.1.a executed"
		set_temp_variable = { temp_opinion = 5 }
		change_fossil_fuel_industry_opinion = yes
		custom_effect_tooltip = MD_escalation_warning_TT
	}

	option = { # Let it be
		name = escalations.1.b
		log = "[GetDateText]: [This.GetName]: bashkiriya.1.b executed"
		set_temp_variable = { temp_opinion = -15 }
		change_fossil_fuel_industry_opinion = yes
	}
}
