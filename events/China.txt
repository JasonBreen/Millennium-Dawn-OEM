	desc = china_sar.23.d

	picture = GFX_politics_talks

	is_triggered_only = yes

	option = {
		name = china_sar.23.a
		log = "[GetDateText]: [This.GetName]: china_sar.23.a executed"
		add_opinion_modifier = {
			target = CHI
			modifier = CSSTA
		}
		reverse_add_opinion_modifier = {
			target = TAI
			modifier = CSSTA
		}
		add_opinion_modifier = {
			target = CHI
			modifier = large_commercial_relations
		}
		reverse_add_opinion_modifier = {
			target = TAI
			modifier = large_commercial_relations
		}
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = CHI_TAI_shared_modifier } } }
			add_dynamic_modifier = { modifier = CHI_TAI_shared_modifier }
		}
		custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = CHI_TAI_shared_modifier }
		add_to_variable = { CHI_TAI_democratic_drift = -0.02 tooltip = democratic_drift_tt }
		add_to_variable = { CHI_TAI_civilian_factories_productivity = 0.03 tooltip = civilian_factories_productivity_tt }
		set_country_flag = TAI_CSSTA_signed
		CHI = { add_to_variable = { CHI_TAI_trust = 10 } }
	}
	option = {
		name = china_sar.23.b
		log = "[GetDateText]: [This.GetName]: china_sar.23.b executed"
		CHI = {
			add_to_variable = { CHI_TAI_trust = -5 }
			if = {
				limit = { check_variable = { CHI_TAI_trust < 0 } }
				set_variable = { CHI_TAI_trust = 0 }
			}
		}
	}
}
